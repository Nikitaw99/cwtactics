 10     /**
 11 	 * Command stack is a class for holding commands. It allows a revision based
 12 	 * handling of the commands including moving between single revisions etc.
 13 	 *
 14 	 * @class
 15 	 */
 16     meowEngine.CommandStack =
 17         meowEngine.Class( /** @lends meowEngine.CommandStack# */ {
 18 
 19 		constructor : function(){
 20 			this._commands = [];
 21 			this._pointer = 0;
 22 		},
 23 
 24 		/**
 25 		 * Commits a command to the command stack and executes it, if wanted
 26 		 * by the executor of this function. If the command stack is not at
 27 		 * the head state, this function will first goto the head revision and
 28 		 * execute all commands, before it processes the command. If you want
 29 		 * to drop all changes between the current revision and the head
 30 		 * revision, then use markHead() first.
 31 		 *
 32 		 * @param {Command} cmd command instance, that will be added to the
 33 		 *					stack
 34 		 * @param {Boolean} execute if true, the command will executed, all
 35 		 *					otherwise ( like undefined ) it won't be executed
 36 		 */
 37 		commit : function( cmd , execute )
 38 		{
 39 
 40 			// commits will always done after the active head state
 41 			if( this._pointer < this._commands.length-1 ){
 42 				this._goToRevision( this._commands.length-1 );
 43 			}
 44 
 45 			if( cmd instanceof meowEngine.Command ){
 46 
 47 				// execute command, if necessary
 48 				if( execute )
 49 					cmd.execute();
 50 
 51 				this._commands.push(cmd);
 52 				this._pointer = this._commands.length -1;
 53 			}
 54 			else throw "Command is not a instance of Command";
 55 		},
 56 
 57 		/**
 58 		 * Brings the state to a given revision. If the target revision is
 59 		 * before the active revision, all revisions between same will be
 60 		 * undone. If the target revision is after the active revision, all
 61 		 * revisions between them will be executed.
 62 		 *
 63 		 * @param {Number} stateNum revision number
 64 		 */
 65 		gotoRev : function( stateNum ){
 66 
 67 			var cmdList = this._commands;
 68 			if( stateNum >= 0 && stateNum < this._commands.length ){
 69 
 70 				if( stateNum < this._pointer ){
 71 
 72 					// undo commands
 73 					while( this._pointer > stateNum ){
 74 						cmdList[ this._pointer ].undo();
 75 						this._pointer--;
 76 					}
 77 				}
 78 				else{
 79 
 80 					// redo commands
 81 					while( this._pointer < stateNum ){
 82 						this._pointer++;
 83 						cmdList[ this._pointer ].execute();
 84 					}
 85 				}
 86 			}
 87 			else{
 88 
 89 			}
 90 
 91 			cmdList = null;
 92 		},
 93 
 94 		/**
 95 		 * Goes strictly to the active head revision. All revisions between the
 96 		 * active revision and the head revision will be executed.
 97 		 */
 98 		gotoHead : function(){
 99 			if( this._pointer < this._commands.length-1 ){
100 				this.goToRevision( this._commands.length-1 );
101 			}
102 		},
103 
104 		/**
105 		 * @return {Number} active revision
106 		 */
107 		rev : function(){
108 			return this._pointer;
109 		},
110 
111 		/**
112 		 * @return {Number} head revision
113 		 */
114 		headRev : function(){
115 			return this._commands.length-1;
116 		},
117 
118 		/**
119 		 * Marks the current active revision as the head revision and drops
120 		 * all revisions after the new head revision from the timeline.
121 		 */
122 		markHead : function(){
123 			this._commands.splice( this._pointer+1,
124 								   this._commands.length -this._pointer -1 );
125 		}
126 	});
127 
128 	/**
129 	 * Commands represents the command pattern. Allows to define a redoable
130 	 * action that provides a do and undo function.
131 	 *
132 	 * @class
133 	 */
134 	meowEngine.Command = meowEngine.Class( /** @lends meowEngine.Command# */ {
135 
136 		/**
137 		 * Executes the command.
138 		 */
139 		execute : meowEngine.EMPTY_FUNCTION,
140 
141 		/**
142 		 * Undoes the command.
143 		 */
144 		undo : meowEngine.EMPTY_FUNCTION
145 	});

147     /**
148 	 *
149 	 * @example
150 	 * // construct interpreter like this
151 	 * // the array as argument is optinal
152 	 * var ipt = new meow.Interpreter([
153 	 *		"c1",function(ctx){},
154 	 *		"c2",function(ctx){ console.log( ctx.x ); },
155 	 *		"c3",function(ctx){}
156 	 * ]);
157 	 *
158 	 * // later you can modify the contexts
159 	 * ipt.putKey( "c4" , function(ctx){} );
160 	 * ipt.removeKey( "c4");
161 	 *
162 	 * // interprete
163 	 * ipt.interprete( meow.toJSON({ _iCode = "c2" , x:10, y:20 , z:30 }) );
164 	 *
165 	 * @class
166 	 */
167 	meowEngine.Interpreter =
168         meowEngine.Class( /** @lends meowEngine.Interpreter# */ {
169 
170         /**
171          *
172          */
173 		constructor : function( contexts ){
174 
175 			var map = {};
176 
177 			if( contexts ){
178 				assert.isArray(contexts);
179                 
180 
181 				var e = contexts.length;
182 				for( var i = 0 ; i < e ; i+=2 ){
183 					this.putKey( contexts[i], contexts[i+1] );
184 				}
185 			}
186 
187 			this._ctxMap = map;
188 		},
189 
190         /**
191          * Puts an inte2rpreter function into the interpreter.
192          *
193          * @param {String} key key of the interpreter function
194          * @param {Function} f interpreter function
195          */
196 		putKey : function( key , f ){
197 
198 			if( typeof key !== 'string' || typeof f !== 'function' ||
199                 typeof this._ctxMap[key] !== 'undefined' ){
200 
201                 throw "Key:"+key+" already exists";
202             }
203 
204 			this._ctxMap[ key ] = f;
205 		},
206 
207         /**
208          * Removes an interpreter function from interpreter.
209          *
210          * @param {String} key key, that will be removed from interpreter
211          */
212 		removeKey : function( key ){
213 
214             if( typeof key !== 'string' ||
215                 typeof this._ctxMap[key] === 'undefined' ){
216                 
217                 throw "Key:"+key+" does not exists";
218             }
219 
220 			delete this._ctxMap[key];
221 		},
222 
223         /**
224          * Interpretes a message.
225          *
226          * @param {String} msg message, that will be interpreted
227          * @return result of the interpreter function
228          */
229 		interprete : function( msg ){
230 
231 			var obj = meowEngine.parseJSON( msg );
232 			var code = obj._iCode;
233 
234             if( typeof this._ctxMap[code] === 'undefined' ){
235                 throw "No correct interpreter for code:"+code+" exists";
236             }
237 
238 			return this._ctxMap[ code ]( obj );
239 		}
240 	});
