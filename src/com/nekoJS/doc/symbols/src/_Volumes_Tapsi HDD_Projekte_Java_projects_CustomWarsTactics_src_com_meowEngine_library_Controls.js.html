  2

 10     
 11 	var _downKeys = {};
 12 	var _actions = {};
 13 
 14 	var _on = 1;
 15 	var _off = 0;
 16 
 17     /** @namespace */
 18     meowEngine.input = {};
 19 
 20     /**
 21      * Registeres a key down event in the meowEngine control system.
 22      *
 23      * @param {Number} keyID key identical number
 24      */
 25 	meowEngine.input.keyDown = function( keyID ){
 26 		_downKeys[keyID] = _on;
 27 	};
 28 
 29     /**
 30      * Registeres a key up event in the meowEngine control system.
 31      *
 32      * @param {Number} keyID key identical number
 33      */
 34 	meowEngine.input.keyUp = function( keyID ){
 35 		// dont delete keys, because normally they
 36 		// will be invoked later again
 37 		_downKeys[keyID] = _off;
 38 	};
 39 
 40     /**
 41      * Binds a key with an action.
 42      *
 43      * @param {Number} keyID key identical number
 44      * @param {String} action action name
 45      */
 46 	meowEngine.input.bindKey = function( keyID, action ){
 47 
 48 		if( typeof action !== 'string' ){
 49 			throw "Input, bindKey has an illegal argument";
 50 		}
 51 
 52 		if( !_actions[action] ){
 53 			_actions = new Array();
 54 		}
 55 		else if( _actions[action].indexOf(keyID) != -1 ){
 56 			return;
 57 		}
 58 
 59 		_actions[action].push(keyID);
 60 	};
 61 
 62     /**
 63      * Unbinds a key from an action.
 64      *
 65      * @param {Number} keyID key identical number
 66      * @param {String} action action name
 67      */
 68 	meowEngine.input.unbindKey = function( keyID, action ){
 69 
 70 		assert.isString(action);
 71 		var array = _actions[action];
 72 
 73 		if( !array ){
 74 			throw "action is undefined or empty, both normally can't be!";
 75 		}
 76 
 77 		var i = array.indexOf(keyID);
 78 		if( i != -1 ){
 79 			array.splice( i , 1 );
 80 		}
 81 
 82 		// delte array if empty
 83 		if( array.length == 0 ){
 84 			delete _actions[action];
 85 		}
 86 		array = null;
 87 	};
 88 
 89     /**
 90      * Checks the status of a key ID.
 91      *
 92      * @return true, if the key is down, else false
 93      */
 94     meowEngine.input.key = function( keyID ){
 95         
 96         return _downKeys[ keyID ] == _on;
 97 	};
 98 
 99     /**
100      * Chechks the status of the given action.
101      *
102      * @return true, if the action is active, else false
103      */
104 	meowEngine.input.action = function( action ){
105 
106 		if( typeof action !== 'string' ){
107 			throw "Input, isActive has an illegal argument";
108 		}
109 
110 		var array = _actions[action];
111 
112 		if( !array ){
113 			throw "action is undefined or empty, both normally can't be!";
114 		}
115 
116 		var i,e = array.length;
117 		for( i = 0; i < e ; i++ ){
118 
119 			// if one is triggered, action is true
120 			if( _downKeys[ array[i] ] ){
121 				return true;
122 			}
123 		}
124 
125 		return false;
126 	};
127 
128     // shortcuts
129     meowEngine.signals.onSignal( meowEngine.MEOW_SHORTCUTS , function(){
130 
131        $inputAction = meowEngine.input.action;
132     });
133 
134 })();
