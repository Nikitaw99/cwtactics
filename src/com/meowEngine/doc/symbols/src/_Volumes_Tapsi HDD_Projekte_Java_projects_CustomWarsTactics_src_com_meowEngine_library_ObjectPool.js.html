1 (function(){
  2

 10     /**
 11      * @class
 12      */
 13     meowEngine.ObjectPool = meowEngine.Class( 
 14                                         /** @lends meowEngine.ObjectPool# */ {
 15 
 16 
 17 		/**
 18 		 *
 19 		 * @param classObj class object, that will be used in this object pool.
 20 		 * @param size Determines the size of the cache pool, if -1, the pool
 21 		 *			   won't have a maximum size.
 22 		 * @param cleanMethod the clean method, that will be called on acquire.
 23 		 *					  If this function is null, no cleaning step will
 24 		 *					  be done.
 25 		 */
 26 		constructor : function( classObj, size, cleanMethod )
 27 		{
 28 			if( classObj == null || typeof size !== 'number' ){
 29 				throw "ObjectPool, illegal argument";
 30 			}
 31 
 32 			if( size < -1 ){
 33 				throw "Illegal size argument";
 34 			}
 35 
 36 			this._clazz = classObj;
 37 			this._pool = [];
 38 			this._size = size;
 39 			this._clean = ( typeof cleanMethod === 'function')?
 40 													cleanMethod : null;
 41 		},
 42 
 43 		/**
 44 		 * Acquires a new object from the pool. If no object is cached, a new
 45 		 * one will be created. Remember, that the constructor is called with
 46 		 * no arguments. The cached class should use a empty constructor and
 47 		 * a cleaning method, if possible.
 48 		 *
 49 		 * @augments arguments are used as arguments for the cleaning method
 50 		 */
 51 		acquire : function()
 52 		{
 53 			var obj;
 54 
 55 			if( this._pool.length > 0 ){
 56 				// use chached instance
 57 				obj = this._pool.pop();
 58 			}
 59 			else{
 60 				// create new instance
 61 				obj = new this._clazz();
 62 			}
 63 
 64 			// call cleaning method if necessary
 65 			if( this._clean != null ){
 66 				this._clean.apply( obj , arguments );
 67 			}
 68 
 69 			return obj;
 70 		},
 71 
 72 		/**
 73 		 * Releases an object back to the pool. If the pool has a maximum size
 74 		 * and the pool have the length of that maximum size, then the object
 75 		 * will not be cached.
 76 		 *
 77 		 * @param obj object that will be pushed into the pool
 78 		 */
 79 		release : function( obj )
 80 		{
 81 			if( !(obj instanceof this._clazz ) ){
 82 				throw "ObjectPool, releasing failed, illegal argument";
 83 			}
 84 
 85 			this._pool.push( obj );
 86 		},
 87 
 88 		/**
 89 		 * @return true if the pool is full, else false.
 90 		 */
 91 		isFull : function()
 92 		{
 93 			return ( this._size == -1 )?
 94 							false : this._pool.length == this._size;
 95 		}
 96 	});
 97    
 98 })();
