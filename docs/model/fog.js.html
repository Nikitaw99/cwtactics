Contains the fog data map. A value 0 means a tile is not visible. A value greater than 0 means it is visible for n units ( n = fog value of the tile ).

util.scoped(function(){
  
  function modifyVisionAt( x,y, pid, range, value ){
    if( !controller.configValue("fogEnabled") ) return;
    
    controller.prepareTags( x, y );
    range = controller.scriptedValue( pid,"vision", range );
    
    var mH = model.mapHeight;
    var mW = model.mapWidth;
    
    var lX;
    var hX;
    var lY = y-range;
    var hY = y+range;
    if( lY < 0 ) lY = 0;
    if( hY >= mH ) hY = mH-1;
    for( ; lY<=hY; lY++ ){
      
      var disY = Math.abs( lY-y );
      lX = x-range+disY;
      hX = x+range-disY;
      if( lX < 0 ) lX = 0;
      if( hX >= mW ) hX = mW-1;
      for( ; lX<=hX; lX++ ){
        
        model.fogData[lX][lY] += value;
      }
    }
  };
