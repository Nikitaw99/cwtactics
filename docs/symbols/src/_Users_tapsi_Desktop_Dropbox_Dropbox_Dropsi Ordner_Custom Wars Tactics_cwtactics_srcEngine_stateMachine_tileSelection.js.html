1 /**
  2  * The client selects a target tile in this step. Unlike {@link controller.stateMachine.structure.ACTION_SELECT_TARGET_A}
  3  * and {@link controller.stateMachine.structure.ACTION_SELECT_TARGET_B} this state allows a free selection
  4  * over the map. Normally this state will be invoked by actions with the isTargetValid attribute. 
  5  */
  6 controller.stateMachine.structure.ACTION_SELECT_TILE = {
  7     
  8     action: function( ev,x,y ){      
  9       if( this.data.actionObject.isTargetValid( this.data, x,y) ){
 10         
 11         this.data.setSelectionTarget(x,y);
 12         return "FLUSH_ACTION";
 13       }
 14       else return this.BREAK_TRANSITION;
 15     },
 16 
 17     cancel: function(){
 18       this.data.setSelectionTarget(-1,-1);
 19       return this.lastState;
 20     }
 21   
 22 };
