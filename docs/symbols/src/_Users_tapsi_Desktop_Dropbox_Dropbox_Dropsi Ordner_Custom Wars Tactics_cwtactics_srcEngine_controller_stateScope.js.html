  9    /**
 10     * Sets a position in the data object. The target name of the position will
 11     * be determined by the tags array.
 12     * 
 13     * @private
 14     * @param {Array} tags list of the position property names
 15     * @param {Number} x x coordinate
 16     * @param {Number} y y coordinate
 17     */
 18   setPosition_: function( tags, x,y ){
 19     var refObj;
 20 
 21     this[tags[0]] = x;
 22     this[tags[1]] = y;
 23     var isValid = (x !== -1 && y !== -1);
 24     var inFog = isValid? (model.fogData[x][y] === 0) : false;
 25 
 26     // ----- UNIT -----
 27     refObj = isValid? model.unitPosMap[x][y] : null;
 28     if( isValid && !inFog && refObj !== null ){
 29       this[tags[2]] = refObj;
 30       this[tags[3]] = model.extractUnitId(refObj);
 31     }
 32     else{
 33       this[tags[2]] = null;
 34       this[tags[3]] = -1;
 35     }
 36 
 37     // ----- PROPERTY -----
 38     refObj = isValid? model.propertyPosMap[x][y] : null;
 39     if( isValid && !inFog && refObj !== null ){
 40       this[tags[4]] = refObj;
 41       this[tags[5]] = model.extractPropertyId(refObj);
 42     }
 43     else{
 44       this[tags[4]] = null;
 45       this[tags[5]] = -1;
 46     }
 47   },

206   /**
207    * Sets the value of a position x,y in the selectiond data.
208    * 
209    * @param {Number} x x coordinate
210    * @param {Number} y y coordinate
211    * @param {Number} defValue value that will be set into every cell of the matrix
212    */
213   setSelectionCenter: function( x,y, defValue ){
214     var data = this.selectionData;
215     var e = data.length;
216     var cx = x;
217     var cy = y;
218     for (var x = 0; x < e; x++) {
219       for (var y = 0; y < e; y++) {
220         data[x][y] = defValue;
221       }
222     }
223 
224     // right bounds are not important
225     this.selectionCX = Math.max(0, cx - CWT_MAX_SELECTION_RANGE*2);
226     this.selectionCY = Math.max(0, cy - CWT_MAX_SELECTION_RANGE*2);
227   },
228 
229   /**
230    * Returns the value of a position x,y in the selectiond data.
231    * 
232    * @param {Number} x x coordinate
233    * @param {Number} y y coordinate
234    */
235   getSelectionValueAt: function( x,y ){
236     var data = this.selectionData;
237     var cy = this.selectionCY;
238     var cx = this.selectionCX;
239     x = x - cx;
240     y = y - cy;
241     var maxLen = data.length;
242 
243     if( x < 0 || y < 0 || x >= maxLen || y >= maxLen ){
244       return -1;
245     }
246     else return data[x][y];
247   },

300   /**

305   addEntry: function( entry ){
306     if( this.menuSize === this.menu.length ){
307       util.raiseError();
308     }
309 
310     this.menu[ this.menuSize ] = entry;
311     this.menuSize++;

321   /**
322    * Prepares the menu for a given source and target position.
323    */
324   prepareMenu: function(){
325     var commandKeys = Object.keys( controller.actionObjects_ );
326 
327     // ----- UNIT -----
328     var unitActable = true;
329     var selectedUnit = this.sourceUnit;
330     if( selectedUnit === null || selectedUnit.owner !== model.turnOwner ){
331       unitActable = false;
332     }
333     else if( !model.canAct( this.sourceUnitId ) ) unitActable = false;
334 
335     // ----- PROPERTY -----
336     var propertyActable = true;
337     var property = this.sourceProperty;
338     if( selectedUnit !== null ) propertyActable = false;
339     if( property === null || property.owner !== model.turnOwner ){
340       propertyActable = false;
341     }
342 
343     for( var i=0,e=commandKeys.length; i<e; i++ ){
344       var action = controller.getActionObject( commandKeys[i] );
345 
346       // IS USER CALLABLE ACTION ?
347       if( !action.condition ) continue;
348 
349       // PRE DEFINED CHECKERS
350       if( action.unitAction === true && !unitActable ) continue;
351       if( action.propertyAction === true && !propertyActable ) continue;
352 
353       // CHECK CONDITION
354       if( action.condition( this ) ){
355         this.addEntry( commandKeys[i] );
356       }
357     }
358   }
