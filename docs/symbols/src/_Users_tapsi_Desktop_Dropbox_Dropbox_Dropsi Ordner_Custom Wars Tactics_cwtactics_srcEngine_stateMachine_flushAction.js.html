1 /**
  2  * Action state that converts the collected action data from client
  3  * to sharable transactions and pushes them into the action stack.
  4  */
  5 controller.stateMachine.structure.FLUSH_ACTION = {
  6   
  7     actionState: function(){
  8       var trapped = false;
  9       if( this.data.movePath !== null ){
 10         var way = this.data.movePath;
 11 
 12         var cx = this.data.sourceX;
 13         var cy = this.data.sourceY;
 14         for( var i=0,e=way.length; i<e; i++ ){
 15 
 16           switch( way[i] ){
 17             case model.MOVE_CODE_DOWN  : cy++; break;
 18             case model.MOVE_CODE_UP    : cy--; break;
 19             case model.MOVE_CODE_LEFT  : cx--; break;
 20             case model.MOVE_CODE_RIGHT : cx++; break;
 21           }
 22 
 23           // ONLY TILES THAT ARE IN FOG OF WAR MUST BE CHECKED AGAINST
 24           // HIDDEN ENEMY UNITS
 25           if( model.fogData[cx][cy] === 0 ){
 26             var unit = model.unitPosMap[cx][cy];
 27             if( unit != null ){
 28 
 29               // TRAPPED ?
 30               if( model.players[model.turnOwner].team !==
 31                     model.players[unit.owner].team ){
 32 
 33                 // CONVERT TO TRAP WAIT
 34                 this.data.action = "TRWT";
 35                 this.data.setTarget(cx,cy);
 36                 way.splice( i );
 37                 trapped = true;
 38               }
 39             }
 40           }
 41         }
 42       }
 43 
 44       // PUSH A COPY INTO THE COMMAND BUFFER
 45       var action;
 46       var actObj;
 47       var actArgs;
 48 
 49       this.data.cleanMovepath();
 50       
 51       if( this.data.movePath.length > 0 ){
 52         action = "MOVE";
 53         actObj = controller.getActionObject( action );
 54         actArgs = actObj.createDataSet( this.data );
 55         actArgs.push(action);
 56         controller.pushSharedAction.apply( null, actArgs );
 57       }
 58 
 59       action = this.data.action;
 60       actObj = this.data.actionObject;
 61       actArgs = actObj.createDataSet( this.data );
 62       actArgs.push(action);
 63       controller.pushSharedAction.apply( null, actArgs );
 64 
 65       if( !trapped && actObj.multiStepAction ){
 66         this.data.inMultiStep = true;
 67         controller.pushSharedAction.apply( null, ["IVMS"] );
 68         return "MULTISTEP_IDLE";
 69       }
 70       else return "IDLE";
 71     }
 72   
 73 };
