1 controller.userAction({
  2   
  3   name:"captureProperty",
  4 
  5   key:"CTPR",
  6   
  7   unitAction: true,
  8   
  9   condition: function( mem ){
 10     return (
 11       mem.targetProperty !== null && 
 12       model.turnOwner !== mem.targetProperty.owner &&
 13 
 14       ( mem.targetUnit === null || mem.targetUnit === mem.sourceUnit ) &&
 15 
 16       model.sheets.tileSheets[ mem.targetProperty.type ].capturePoints > 0 &&
 17       model.sheets.unitSheets[ mem.sourceUnit.type ].captures > 0
 18     );
 19   },
 20   
 21   createDataSet: function( mem ){
 22     
 23     // ONE POINT FOR EVERY 10 HP STARTING WITH 9
 24     var points = parseInt( mem.sourceUnit.hp/10, 10 ) +1;
 25     
 26     return [
 27       mem.sourceUnitId,
 28       mem.targetPropertyId,
 29       mem.targetX,
 30       mem.targetY,
 31       points
 32     ];
 33   },
 34   
 35   /**
 36    * Captures a property.
 37    *
 38    * @param {Number} cid capturer id
 39    * @param {Number} prid property id
 40    * @param {Number} px x coordinate
 41    * @param {Number} py y coordinate
 42    * @param {Number} points capture points
 43    *
 44    * @methodOf controller.actions
 45    * @name captureProperty
 46    */
 47   action: function( cid, prid, px,py, points ){
 48     var selectedUnit = model.units[cid];
 49     var property = model.properties[prid];
 50     var unitSh = model.sheets.unitSheets[ selectedUnit.type ];
 51 
 52     selectedUnit.ST_CAPTURES = true;
 53     
 54     property.capturePoints -= points;
 55     if( property.capturePoints <= 0 ){
 56       var x = px;
 57       var y = py;
 58 
 59       if( DEBUG ){
 60         util.logInfo( "property at (",x,",",y,") captured");
 61       }
 62 
 63       // ADD VISION
 64       controller.actions.addVision( x,y, model.sheets.tileSheets[property.type].vision );
 65 
 66       if( property.type === 'HQTR' ){
 67         var pid = property.owner;
 68         var oldPlayer = model.players[pid];
 69 
 70         for( var i = pid*CWT_MAX_UNITS_PER_PLAYER,
 71                  e = i+CWT_MAX_UNITS_PER_PLAYER; i<e; i++ ){
 72 
 73           model.units[i].owner = -1;
 74           model.eraseUnitPosition(i);
 75         }
 76 
 77         for( var i = 0, e = model.properties.length; i<e; i++ ){
 78           if( model.properties[i].owner === pid ){
 79             model.properties[i].owner = -1;
 80           }
 81         }
 82 
 83         oldPlayer.team = -1;
 84 
 85         // check win/loose
 86         var _teamFound = -1;
 87         for( var i=0,e=model.players.length; i<e; i++ ){
 88           var player = model.players[i];
 89           if( player.team !== -1 ){
 90 
 91             // FOUND AN ALIVE PLAYER
 92             if( _teamFound === -1 ) _teamFound = player.team;
 93             else if( _teamFound !== player.team ){
 94               _teamFound = -1;
 95               break;
 96             }
 97           }
 98         }
 99 
100         // NO OPPOSITE TEAMS LEFT ?
101         if( _teamFound !== -1 ){
102           controller.pushSharedAction("endGame");
103         }
104       }
105 
106       // set new meta data for property
107       property.capturePoints = 20;
108       property.owner = selectedUnit.owner;
109 
110       var capLimit = model.rules.captureWinLimit;
111       if( capLimit !== 0 && capLimit <= model.countProperties() ){
112         controller.pushSharedAction("endGame");
113       }
114     }
115 
116     controller.actions.wait( cid );
117   }
118   
119 });
