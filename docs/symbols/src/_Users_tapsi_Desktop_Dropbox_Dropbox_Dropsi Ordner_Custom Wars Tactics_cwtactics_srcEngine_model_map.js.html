1 /**
  2  * Map table that holds all known tiles.
  3  */
  4 model.map = util.matrix( CWT_MAX_MAP_WIDTH, CWT_MAX_MAP_HEIGHT, null );
  5 
  6 /**
  7  * Returns the current active map height.
  8  */
  9 model.mapHeight = -1;
 10 
 11 /**
 12  * Returns the current active map width.
 13  */
 14 model.mapWidth = -1;
 15 
 16 /**
 17  * Returns the distance of two positions.
 18  *
 19  * @param {Number} sx
 20  * @param {Number} sy
 21  * @param {Number} tx
 22  * @param {Number} ty
 23  */
 24 model.distance = function( sx,sy,tx,ty ){
 25   var dx = Math.abs(sx-tx);
 26   var dy = Math.abs(sy-ty);
 27   return dx+dy;
 28 };
 29 
 30 /**
 31  * Returns the move code from a tile ax,ay to bx,by.
 32  * 
 33  * @param {Number} ax
 34  * @param {Number} ay
 35  * @param {Number} bx
 36  * @param {Number} by
 37  */
 38 model.moveCodeFromAtoB = function( ax,ay, bx,by ){
 39   if( model.distance( ax,ay, bx,by ) !== 1 ){
 40     util.raiseError("both positions haven't a distance of 1");
 41   }
 42 
 43   // MUST FIT
 44   if( bx < ax ){ return model.MOVE_CODE_LEFT; }
 45   if( bx > ax ){ return model.MOVE_CODE_RIGHT; }
 46   if( by < ay ){ return model.MOVE_CODE_UP; }
 47   if( by > ay ){ return model.MOVE_CODE_DOWN; }
 48 
 49   util.unexpectedSituationError();
 50 };

 52 model.isValidPosition = function( x,y ){
 53   return ( x >= 0 && y >= 0 && x < model.mapWidth && y < model.mapHeight );
 54 };
 55 
 56 /**
 57  * Returns true if an own unit, in relationship to a given player id, is on a
 58  * tile at a given position x,y.
 59  * 
 60  * @param {Number} x
 61  * @param {Number} y
 62  * @param {Number} pid
 63  * @returns {Boolean}
 64  */
 65 model.thereIsAnOwnUnitAt = function( x,y,pid ){
 66   if( !model.isValidPosition(x,y) ) return false;
 67 
 68   var unit = model.unitPosMap[x][y];
 69   return ( unit !== null && pid === unit.owner );
 70 };
 71 
 72 /**
 73  * Returns true if an allied unit, in relationship to a given player id, is on 
 74  * a tile at a given position x,y.
 75  * 
 76  * @param {Number} x
 77  * @param {Number} y
 78  * @param {Number} pid
 79  * @returns {Boolean}
 80  */
 81 model.thereIsAnAlliedUnitAt = function( x,y,pid ){
 82   if( !model.isValidPosition(x,y) ) return false;
 83 
 84   var unit = model.unitPosMap[x][y];
 85   return ( unit !== null && pid !== unit.owner &&
 86             model.players[pid].team === model.players[unit.owner].team);
 87 };

 89 /**
 90  * Returns true if an enemy unit, in relationship to a given player id, is on a
 91  * tile at a given position x,y.
 92  * 
 93  * @param {Number} x
 94  * @param {Number} y
 95  * @param {Number} pid
 96  * @returns {Boolean}
 97  */
 98 model.thereIsAnEnemyUnitAt = function( x,y,pid ){
 99   if( !model.isValidPosition(x,y) ) return false;
100 
101   var unit = model.unitPosMap[x][y];
102   return ( unit !== null && pid !== unit.owner &&
103             model.players[pid].team !== model.players[unit.owner].team );
104 };
