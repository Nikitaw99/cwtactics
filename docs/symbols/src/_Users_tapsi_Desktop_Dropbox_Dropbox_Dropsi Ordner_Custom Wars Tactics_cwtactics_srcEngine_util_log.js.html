1 /**
  2  * Raises an error in the active Javascript environment.
  3  *
  4  * @param {...Object} reason A number of arguments that will be used as error message.
  5  *                           If an argument isn't a String then it will be converted to
  6  *                           String by the toString() function.
  7  */
  8 util.raiseError = function( reason ){
  9   
 10   if( arguments.length === 0 ){
 11     reason = "CustomWars Debug:: An error was raised";
 12   }
 13   else if( arguments.length > 1 ){
 14     reason = Array.prototype.join.call( arguments, " " );
 15   }
 16     
 17   throw Error( reason );
 18 };
 19 
 20 /**
 21  * Logging function.
 22  *
 23  * @param {...Object} reason A number of arguments that will be used as message.
 24  *                           If an argument isn't a String then it will be converted to
 25  *                           String by the toString() function.
 26  * @config
 27  */
 28 util.log = function( msg ){
 29   if( arguments.length > 1 ){
 30     msg = Array.prototype.join.call( arguments, " " );
 31   }
 32 
 33   console.log( msg );
 34 }
 35 
 36 /**
 37  * Overwritable logging function.
 38  *
 39  * @deprecated will be removed in version 0.3
 40  */
 41 util.logInfo = function(){
 42   util.log.apply( this, arguments );
 43 };
