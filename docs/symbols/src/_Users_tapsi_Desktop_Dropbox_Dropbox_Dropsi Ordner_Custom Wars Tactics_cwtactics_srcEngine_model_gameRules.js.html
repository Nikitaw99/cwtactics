 48 /**
 49  * Unites modification rules and custom rules to a valid game round rule
 50  * object.
 51  */
 52 model.setRulesByOption = function( options ){
 53   var modRules = model.sheets.defaultRules;
 54   var rules = model.rules;
 55 
 56   var keys = Object.keys( modRules );
 57   for( var i=0,e=keys.length; i<e; i++ ){
 58 
 59     var key = keys[i];
 60 
 61     // TAKE EITHER MOD DEFAULT VALUES OR CUSTOM VALUES
 62     if( options.hasOwnProperty(key) ){
 63       rules[key] = options[key];
 64     } else{
 65       rules[key] = modRules[key];
 66     }
 67   }
 68 };
