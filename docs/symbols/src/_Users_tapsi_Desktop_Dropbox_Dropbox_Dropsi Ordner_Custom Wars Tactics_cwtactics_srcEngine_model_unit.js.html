1 /**
  2  * Matrix with the same metrics like the map. Every unit is placed into the 
  3  * cell that represents its position.
  4  */
  5 model.unitPosMap = util.matrix( CWT_MAX_MAP_WIDTH, CWT_MAX_MAP_HEIGHT, null );
  6 
  7 /**
  8  * List of all unit objects. An inactive unit is marked with 
  9  * {@link CWT_INACTIVE_ID} as owner.
 10  */
 11 model.units = util.list( CWT_MAX_PLAYER*CWT_MAX_UNITS_PER_PLAYER, function(){
 12   return {
 13     x:0,
 14     y:0,
 15     hp: 99,
 16     ammo: 0,
 17     type: null,
 18     loadedIn: -1,
 19     fuel: 0,
 20     owner: CWT_INACTIVE_ID,
 21 
 22     ST_HP_NUM:0,
 23     ST_LOW_AMMO:false,
 24     ST_LOW_FUEL:false,
 25     ST_HAS_LOADS:false,
 26     ST_CAPTURES: false,
 27 
 28     _clientData_: {}
 29   };
 30 });
 31 
 32 /**
 33  * Extracts the identical number from an unit object.
 34  *
 35  * @param unit
 36  */
 37 model.extractUnitId = function( unit ){
 38   if( unit === null ){
 39     util.raiseError("unit argument cannot be null");
 40   }
 41 
 42   var units = model.units;
 43   for( var i=0,e=units.length; i<e; i++ ){
 44     if( units[i] === unit ) return i;
 45   }
 46 
 47   util.raiseError("cannot find unit", unit );
 48 };

 50 /**
 51  * Returns true if a player with a given player id has free slots for new units.
 52  * 
 53  * @param {Number} pid player id
 54  * @returns {Boolean}
 55  */
 56 model.hasFreeUnitSlots = function( pid ){
 57   var startIndex = pid*CWT_MAX_UNITS_PER_PLAYER;
 58   for( var i=0, e=startIndex+CWT_MAX_UNITS_PER_PLAYER; i<e; i++ ){
 59     if( model.units[i].owner !== CWT_INACTIVE_ID ){ return true; }
 60   }
 61 
 62   return false;
 63 };
 64 
 65 /**
 66  * Returns true if a given position is occupied by an unit, else false.
 67  *
 68  * @param {Number} x x coordinate
 69  * @param {Number} y y coordinate
 70  */
 71 model.tileOccupiedByUnit = function( x,y ){
 72   var unit = model.unitPosMap[x][y];
 73   if( unit === null ) return false;
 74   else return model.extractUnitId( unit );
 75 };

 77 /**
 78  * Counts all units that are owned by the player with the given player id.
 79  *
 80  * @param {Number} pid player id
 81  */
 82 model.countUnits = function( pid ){
 83   var startIndex = pid*CWT_MAX_UNITS_PER_PLAYER;
 84   var n = 0
 85   for( var i=0, e=startIndex+CWT_MAX_UNITS_PER_PLAYER; i<e; i++ ){
 86     if( model.units[i].owner !== CWT_INACTIVE_ID ){
 87       n++;
 88     }
 89   }
 90 
 91   return n;
 92 };
