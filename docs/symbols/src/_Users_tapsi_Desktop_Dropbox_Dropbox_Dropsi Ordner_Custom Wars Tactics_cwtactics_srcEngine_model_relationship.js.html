1 /**
  2  * Two objects which have the same owner.
  3  *
  4  * @constant
  5  * @deprecated will be removed in version 0.3
  6  */
  7 model.RELATIONSHIP_SAME_OWNER = 0;
  8 
  9 /**
 10  * Two objects which have differnt of the same team.
 11  *
 12  * @constant
 13  * @deprecated will be removed in version 0.3
 14  */
 15 model.RELATIONSHIP_ALLIED = 1;
 16 
 17 /**
 18  * Two objects which have differnt owners of different teams.
 19  *
 20  * @constant
 21  * @deprecated will be removed in version 0.3
 22  */
 23 model.RELATIONSHIP_ENEMY = 2;
 24 
 25 /**
 26  * Two objects which have no relationship because one or both of them
 27  * hasn't an owner.
 28  *
 29  * @constant
 30  * @deprecated will be removed in version 0.3
 31  */
 32 model.RELATIONSHIP_NONE = 3;
 33 
 34 /**
 35  * @constant
 36  * @deprecated will be removed in version 0.3
 37  */
 38 model.RELATIONSHIP_SAME_OBJECT = 4;
 39 
 40 
 41 /**
 42  * Returns the relationship between two player identicals.
 43  *
 44  * @deprecated will be removed in version 0.3
 45  *
 46  * @param pidA player id or ownable object
 47  * @param pidB player id or ownable object
 48  */
 49 model.relationshipBetween = function( pidA, pidB ){
 50   if( pidA === null || pidB === null ){
 51     return model.RELATIONSHIP_NONE;
 52   }
 53 
 54   if( typeof pidA !== 'number' &&  typeof pidB !== 'number'
 55     && pidA === pidB ) return model.RELATIONSHIP_SAME_OBJECT;
 56 
 57   if( typeof pidA !== 'number' ) pidA = pidA.owner;
 58   if( typeof pidB !== 'number' ) pidB = pidB.owner;
 59 
 60   if( pidA === pidB ) return model.RELATIONSHIP_SAME_OWNER;
 61   else {
 62     var tidA = model.players[ pidA ];
 63     var tidB = model.players[ pidB ];
 64     if( tidA === tidB ){
 65       return model.RELATIONSHIP_ALLIED;
 66     }
 67     else return model.RELATIONSHIP_ENEMY;
 68   }
 69 };
