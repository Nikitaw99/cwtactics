1 /**
  2  * List of all available properties of a game round. If a property is not 
  3  * used it will be marked with an owner value {@link CWT_INACTIVE_ID}.
  4  */
  5 model.properties = util.list( CWT_MAX_PROPERTIES+1, function(){
  6   return {
  7     capturePoints: 20,
  8     owner: -1,
  9     type: null
 10   };
 11 });
 12 
 13 /**
 14  * Matrix that has the same metrics as the game map. Every property will be 
 15  * placed in the cell that represents its position. A property will be 
 16  * accessed by model.propertyPosMap[x][y].
 17  */
 18 model.propertyPosMap = util.matrix(
 19   CWT_MAX_MAP_WIDTH,
 20   CWT_MAX_MAP_HEIGHT,
 21   null
 22 );
 23 
 24 /**
 25  * Returns true if the tile at position x,y is a property, else false.
 26  * 
 27  * @param {Number} x x coordinate
 28  * @param {Number} y y coordinate
 29  */
 30 model.tileIsProperty = function( x,y ){
 31   var prop = model.propertyPosMap[x][y];
 32   return prop !== undefined;
 33 };
 34 
 35 /**
 36  * Extracts the identical number from a property object.
 37  *
 38  * @param property
 39  */
 40 model.extractPropertyId = function( property ){
 41   if( property === null ){
 42     util.raiseError("property argument cannot be null");
 43   }
 44 
 45   var props = model.properties;
 46   for( var i=0,e=props.length; i<e; i++ ){
 47     if( props[i] === property ) return i;
 48   }
 49 
 50   util.raiseError("cannot find property",property );
 51 };
 52 
 53 /**
 54  * Counts all properties owned by the player with the given player id.
 55  *
 56  * @param {Number} pid player id
 57  */
 58 model.countProperties = function( pid ){
 59 
 60   var n = 0;
 61   var props = model.properties;
 62   for( var i=0,e=props.length; i<e; i++ ){
 63     if( props[i].owner === pid ){
 64       n++;
 65     }
 66   }
 67 
 68   return n;
 69 };
