1 /**
  2  * List that contains all player instances. An inactive player is marked 
  3  * with {@link CWT_INACTIVE_ID} as team number.
  4  */
  5 model.players = util.list( CWT_MAX_PLAYER+1, function( index ){
  6   var neutral = (index === CWT_MAX_PLAYER );
  7   return {
  8     gold: 0,
  9     team: ( neutral )? 9999 : CWT_INACTIVE_ID,
 10     name: ( neutral )? "NEUTRAL" : null
 11   };
 12 });
 13 
 14 /**
 15  * Returns true if the given id is a neutral player, else false.
 16  * 
 17  * @param {Number} id player id
 18  * @deprecated will be removed with version 0.3 because the neutral player will
 19  *             be dropped.
 20  */
 21 model.isNeutralPlayer = function( id ){
 22   return model.neutralPlayerId === id;
 23 };
 24 
 25 /**
 26  * Extracts the identical number from an player object.
 27  *
 28  * @param player
 29  */
 30 model.extractPlayerId = function( player ){
 31   if( player === null ){
 32     util.raiseError("player argument cannot be null");
 33   }
 34 
 35   var players = model.players;
 36   for( var i=0,e=players.length; i<e; i++ ){
 37     if( players[i] === player ) return i;
 38   }
 39 
 40   util.raiseError( "cannot find player", players );
 41 };
 42 
 43 /**
 44  * Returns the neutral player id.
 45  * 
 46  * @deprecated will be dropped in version 0.3
 47  */
 48 model.neutralPlayerId = model.players.length-1;
 49 
 50 /**
 51  * Returns true if player id A is in the same team 
 52  * as player id B, else false. 
 53  * 
 54  * @param {Number} pidA player id
 55  * @param {Number} pidB player id
 56  */
 57 model.alliedPlayers = function( pidA, pidB ){
 58   return model.players[pidA].team === model.players[pidB].team;
 59 };
 60 
 61 /**
 62  * Returns true if player id A is not in the same 
 63  * team as player id B, else false. 
 64  * 
 65  * @param {Number} pidA player id
 66  * @param {Number} pidB player id
 67  */
 68 model.enemyPlayers = function( pidA, pidB ){
 69   return model.players[pidA].team !== model.players[pidB].team;
 70 };
