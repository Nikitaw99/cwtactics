1 controller.engineAction({
  2 
  3   name:"siloRegeneration",
  4   
  5   key:"SIRE",
  6   
  7   /**
  8    * Invokes a day tick for an empty silo.
  9    *
 10    * @param {Number} pid property id
 11    *
 12    * @methodOf controller.actions
 13    * @name siloRegeneration
 14    */
 15   action: function( pid ){
 16     
 17     var maxDays = model.rules.siloRegeneration;
 18     if( maxDays === -1 ) return;
 19     
 20     if( model.regeneratingSilos.hasOwnProperty(pid) ){
 21       model.regeneratingSilos[pid]++;
 22     }
 23     else model.regeneratingSilos[pid] = 1;
 24     
 25     if( model.regeneratingSilos[pid] >= maxDays ){
 26       delete model.regeneratingSilos[pid];
 27       model.properties[pid].type = "SILO";
 28     }
 29   }
 30 });
