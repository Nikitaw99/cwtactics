1 controller.engineAction({
  2 
  3   name: "moveUnit",
  4 
  5   key: "MOVE",
  6 
  7   shared: true,
  8 
  9   createDataSet: function( data ){
 10     return [ data.movePath, data.sourceUnitId, data.sourceX, data.sourceY ];
 11   },
 12   
 13   /**
 14    * Moves an unit from A to B.
 15    *
 16    * @param {Array} way move path of the unit
 17    * @param {Number} uid moving unit id 
 18    * @param {Number} x x coordinate of the source
 19    * @param {Number} y y coordinate of the source
 20    *
 21    * @methodOf controller.actions
 22    * @name moveUnit
 23    */
 24   action: function( way, uid, x,y ){
 25     var cX = x;
 26     var cY = y;
 27     var unit = model.units[ uid ];
 28     var uType = model.sheets.unitSheets[ unit.type ];
 29     var mType = model.sheets.movetypeSheets[ uType.moveType ];
 30 
 31     // CHECK MOVE WAY END
 32     var lastIndex = way.length-1;
 33     var fuelUsed = 0;
 34     for( var i=0,e=way.length; i<e; i++ ){
 35 
 36       // GET NEW CURRENT POSITION
 37       switch( way[i] ){
 38 
 39         case model.MOVE_CODE_UP:
 40           if( cY === 0 ) util.logError(
 41             "cannot do move command UP because",
 42             "current position is at the border"
 43           );
 44           cY--;
 45           break;
 46 
 47         case model.MOVE_CODE_RIGHT:
 48           if( cX === model.mapWidth-1 ) util.logError(
 49             "cannot do move command RIGHT because",
 50             "current position is at the border"
 51           );
 52           cX++;
 53           break;
 54 
 55         case model.MOVE_CODE_DOWN:
 56           if( cY === model.mapHeight-1 )util.logError(
 57             "cannot do move command DOWN because",
 58             "current position is at the border"
 59           );
 60           cY++;
 61           break;
 62 
 63         case model.MOVE_CODE_LEFT:
 64           if( cX === 0 ) util.logError(
 65             "cannot do move command LEFT because",
 66             "current position is at the border"
 67           );
 68           cX--;
 69           break;
 70 
 71         default: util.logError("unknown command ",way[i]);
 72       }
 73 
 74       // IS WAY BLOCKED ? TODO
 75       if( false && model.isWayBlocked( cX, cY, unit.owner, i == e-1 ) ){
 76 
 77         lastIndex = i-1;
 78 
 79         // GP BACK
 80         switch( way[i] ){
 81 
 82           case model.MOVE_CODE_UP:
 83             cY++;
 84             break;
 85 
 86           case model.MOVE_CODE_RIGHT:
 87             cX--;
 88             break;
 89 
 90           case model.MOVE_CODE_DOWN:
 91             cY--;
 92             break;
 93 
 94           case model.MOVE_CODE_LEFT:
 95             cX++;
 96             break;
 97         }
 98 
 99 
100         if( lastIndex == -1 ){
101 
102           // THAT IS A FAULT
103           util.raiseError( "unit is blocked by an enemy, but the enemy stands beside the start tile, that is a logic fault!" );
104         }
105 
106         break;
107       }
108 
109       // INCREASE FUEL USAGE
110       fuelUsed += model.moveCosts( mType, model.map[cX][cY] );
111     }
112 
113     unit.fuel -= fuelUsed;
114     
115     var uSheet = model.sheets.unitSheets[ unit.type ];
116     var cFuel = unit.fuel;
117     var mFuel = uSheet.maxFuel;
118     if( cFuel < parseInt(mFuel*0.25, 10) ) unit.ST_LOW_FUEL = true;
119     else                                   unit.ST_LOW_FUEL = false;
120     unit.ST_CAPTURES = false;
121 
122     // DO NOT ERASE POSITION IF UNIT WAS LOADED OR HIDDEN (NOT INGAME HIDDEN)
123     // SOMEWHERE
124     if( unit.x !== -1 && unit.y !== -1 ){
125       model.unitPosMap[ unit.x ][ unit.y ] = null;
126       unit.x = -1;
127       unit.y = -1;
128       controller.actions.removeVision( unit.x, unit.y, model.sheets.unitSheets[ unit.type ].vision );
129     }
130 
131     // DO NOT SET NEW POSITION IF THE POSITION IS OCCUPIED
132     // THE SET POSITION LOGIC MUST BE DONE BY THE ACTION
133     if( model.unitPosMap[cX][cY] === null ){
134       unit.x = cX;
135       unit.y = cY;
136       model.unitPosMap[ cX ][ cY ] = unit;
137       controller.actions.addVision(  cX, cY, model.sheets.unitSheets[ unit.type ].vision );
138     }
139 
140     if( DEBUG ){
141       util.log( "moved unit",uid,"from (",x,",",y,") to (",cX,",",cY,")" );
142     }
143   }
144 
145 })
