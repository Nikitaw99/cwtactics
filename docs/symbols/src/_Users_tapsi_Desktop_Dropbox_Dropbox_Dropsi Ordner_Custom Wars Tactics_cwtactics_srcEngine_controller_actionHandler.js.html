1 /**
  2  * Wrapper object to make actions locally callable without invoking a transaction context nor being
  3  * evaluated over the action stack.
  4  * 
  5  * Note that the client expects actions to be invoked over the action stack. Only smaller actions like
  6  * wait should be invoked directly, all other things should be pushed into the action stack.
  7  * 
  8  * @namespace
  9  */
 10 controller.actions = {};
 11 
 12 /** 
 13  * Contains all known action objects.
 14  *  
 15  * @private 
 16  */
 17 controller.actionObjects_ = {};
 18 
 19 /**
 20  * Registers an internal non user callable action.
 21  *
 22  * @param impl action implementation
 23  */
 24 controller.engineAction = function( impl ){
 25   
 26   // CHECKS
 27   if( !impl.hasOwnProperty("key") ) util.raiseError("action key isn't defined");
 28   if( controller.actionObjects_.hasOwnProperty( impl.key ) ) util.raiseError("action key is already registered");
 29   if( !impl.hasOwnProperty("name") ) util.raiseError("action name isn't defined");
 30   if( !impl.hasOwnProperty("action") ) util.raiseError("action implementation isn't defined");
 31   if( !impl.hasOwnProperty("condition") ) impl.condition = false;
 32   if( !impl.hasOwnProperty("shared") ) impl.shared = false;
 33   
 34   // REGISTER PROGRAMATIC LINK
 35   var key = impl.key;
 36   controller.actions[ impl.name ] = function(){
 37     var cmd = controller.actionObjects_[ key ];
 38     cmd.action.apply( cmd, arguments );
 39   };
 40   
 41   controller.actionObjects_[ key ] = impl;
 42 };
 43 
 44 /**
 45  * Registers an user callable action.
 46  *
 47  * @param impl action implementation
 48  */
 49 controller.userAction = function( impl ){
 50   
 51   // CHECKS
 52   if( !impl.hasOwnProperty("condition") ) impl.condition = util.FUNCTION_TRUE_RETURNER;
 53   if( !impl.hasOwnProperty("prepareMenu") ) impl.prepareMenu = null;
 54   if( !impl.hasOwnProperty("prepareTargets") ) impl.prepareTargets = null;
 55   if( !impl.hasOwnProperty("isTargetValid") ) impl.isTargetValid = null;
 56   if( !impl.hasOwnProperty("multiStepAction") ) impl.multiStepAction = false;
 57   if( !impl.hasOwnProperty("shared") ) impl.shared = true;
 58   if( impl.prepareTargets !== null && !impl.hasOwnProperty("targetSelectionType") ) impl.targetSelectionType = "A";
 59   
 60   if( !impl.hasOwnProperty("createDataSet") ) util.raiseError("action data set creation handler isn't defined");
 61   if( impl.prepareTargets !== null && impl.isTargetValid !== null ){
 62     util.raiseError("only one selection type can be used in an action");
 63   }
 64   
 65   controller.engineAction( impl );
 66 };
 67 
 68 /**
 69  * A client action is always a locally invoked action.
 70  * 
 71  * @param impl action implementation
 72  */
 73 controller.clientAction = function( impl ){
 74   impl.shared = false;
 75   
 76   controller.userAction( impl );
 77 };
 78 
 79 /**
 80  * Returns an action object for a given action key.
 81  * 
 82  * @param {String} actionKey action key
 83  */
 84 controller.getActionObject = function( actionKey ){
 85   return controller.actionObjects_[actionKey];
 86 };
