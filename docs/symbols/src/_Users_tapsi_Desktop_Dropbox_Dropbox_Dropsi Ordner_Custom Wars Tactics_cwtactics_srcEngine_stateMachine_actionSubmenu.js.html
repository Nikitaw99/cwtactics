1 /**
  2  * Action sub menu state that generates a list of possible sub actions for 
  3  * an action to modify the selected action like select an unit that should
  4  * be unloaded.
  5  */
  6 controller.stateMachine.structure.ACTION_SUBMENU = {
  7   
  8   onenter: function( ev, x,y ){
  9     this.data.cleanMenu();
 10     controller.getActionObject( this.data.action ).prepareMenu( this.data );
 11     if( this.data.menuSize === 0 ){        
 12       util.raiseError("sub menu cannot be empty");
 13     }
 14   },
 15   
 16   action: function( ev, index ){
 17     var action = this.data.menu[ index ];
 18     
 19     if( action === "done" ){
 20       return "IDLE";
 21     }
 22     
 23     this.data.subAction = action;
 24     
 25     if( this.data.actionObject.prepareTargets !== null && 
 26         this.data.actionObject.targetSelectionType === "B" ){
 27       
 28       return this.data.prepareSelection();
 29     }
 30     else return "FLUSH_ACTION";
 31   },
 32   
 33   
 34   cancel: function(){
 35     if( this.data.inMultiStep ) return this.lastState;
 36     
 37     this.data.cleanMenu();
 38     this.data.prepareMenu();
 39     
 40     return this.lastState;
 41   }
 42 }
