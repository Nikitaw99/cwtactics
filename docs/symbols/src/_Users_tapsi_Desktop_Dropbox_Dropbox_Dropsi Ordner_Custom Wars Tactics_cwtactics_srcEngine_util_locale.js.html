1 /** 
  2  * Contains all active languages.
  3  */
  4 util.i18n_data = { en:{} };
  5 
  6 /** 
  7  * The active language object for the game.
  8  *
  9  * @default english
 10  */
 11 util.i18n_lang = util.i18n_data.en;
 12 
 13 /**
 14  * Returns a localized string for a given key or if not exist the key itself.
 15  *
 16  * @param {String} key
 17  */
 18 util.i18n_localized = function( key ){
 19   var result = this.i18n_lang[key];
 20   return ( result === undefined )? key: result;
 21 };
 22 
 23 /**
 24  * Sets the active language.
 25  *
 26  * @param {String} langKey
 27  */
 28 util.i18n_setLanguage = function( langKey ){
 29   if( !util.i18n_data.hasOwnProperty( langKey ) ){
 30     throw Error("unknown language");
 31   }
 32   else{
 33     util.i18n_lang = util.i18n_data[langKey];
 34   }
 35 };
 36 
 37 /**
 38  * Appends data to a given language.
 39  *
 40  * @param {String} langKey
 41  * @param {Object} data
 42  */
 43 util.i18n_appendToLanguage = function( langKey, data ){
 44 
 45   // CREATE LANG NAMESPACE IF NOT EXISTS
 46   if( !util.i18n_data.hasOwnProperty( langKey ) ){
 47     util.i18n_data[ langKey ] = {};
 48   }
 49 
 50   // APPEND NEW KEYS
 51   var langNs = util.i18n_data[ langKey ];
 52   var keys = Object.keys( data );
 53   for( var i=0,e=keys.length; i<e; i++ ){
 54     langNs[ keys[i] ] = data[ keys[i] ];
 55   }
 56 };
