1 /**
  2  * The client selects a target tile in this step. The selected action generates a map of
  3  * selectable tiles. This selection will be invoked before(!) the sub menu.
  4  */
  5 controller.stateMachine.structure.ACTION_SELECT_TARGET_A = {
  6     
  7     action: function( ev,x,y ){
  8       
  9       if( this.data.getSelectionValueAt(x,y) < 0){
 10         if( CLIENT_DEBUG ){
 11           util.log("break event because selection is not in the map");
 12         }
 13 
 14         return this.BREAK_TRANSITION;
 15       }
 16 
 17       this.data.setSelectionTarget(x,y);
 18       return "FLUSH_ACTION";
 19     },
 20 
 21     cancel: function(){
 22       return this.lastState;
 23     }
 24   
 25 };
