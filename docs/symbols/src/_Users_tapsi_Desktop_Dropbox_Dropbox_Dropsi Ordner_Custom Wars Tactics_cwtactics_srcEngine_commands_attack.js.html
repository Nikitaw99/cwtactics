 49   counterWeapon: function( xdef, ydef, xatt, yatt ){
 50     var dis = Math.abs(xdef-xatt) + Math.abs( ydef-yatt );
 51     if( dis > 1 ) return null; // INDIRECT ATTACK
 52 
 53     var def = model.unitPosMap[xdef][ydef];
 54     var att = model.unitPosMap[xatt][yatt];
 55     var defsheet = model.sheets.unitSheets[ def.type ];
 56     var mainWp = defsheet[model.PRIMARY_WEAPON_TAG];
 57     var sideWp = defsheet[model.SECONDARY_WEAPON_TAG];
 58 
 59     if( mainWp !== undefined ){
 60       mainWp = model.sheets.weaponSheets[ mainWp ];
 61       var minR = mainWp.minRange;
 62       var maxR = mainWp.maxRange;
 63       if( minR === 1 && maxR === 1 && dis === 1 ) return mainWp;
 64     }
 65 
 66     if( sideWp !== undefined ){
 67       sideWp = model.sheets.weaponSheets[ sideWp ];
 68       var minR = sideWp.minRange;
 69       var maxR = sideWp.maxRange;
 70       if( minR === 1 && maxR === 1 && dis === 1 ) return sideWp;
 71     }
 72 
 73     // NO POSSIBLE COUNTER WEAPON
 74     return null;

283    * One unit attacks another unit.

295   action: function( aid, admg, aUseAmmo, did, ddmg, dUseAmmo ){
296     
297     // ATTACK
298     controller.actions.damageUnit( did, admg );
299     controller.actions.wait( aid );
300     
301     if( aUseAmmo ){
302       var att = model.units[aid];
303       var uSheet = model.sheets.unitSheets[ att.type ];
304       var cAmmo = att.ammo;
305       var mAmmo = uSheet.maxAmmo;
306       if( cAmmo <= parseInt(mAmmo*0.25, 10) ) att.ST_LOW_AMMO = true;
307       else                                    att.ST_LOW_AMMO = false;
308       if( mAmmo === 0 )                       att.ST_LOW_AMMO = false;
309     }
310     
311     // COUNTER ATTACK
312     if( model.units[did].owner !== CWT_INACTIVE_ID ){
313       controller.actions.damageUnit( aid, ddmg );
314       
315       if( dUseAmmo ){
316         var def = model.units[did];
317         var uSheet = model.sheets.unitSheets[ def.type ];
318         var cAmmo = def.ammo;
319         var mAmmo = uSheet.maxAmmo;
320         if( cAmmo <= parseInt(mAmmo*0.25, 10) ) def.ST_LOW_AMMO = true;
321         else                                    def.ST_LOW_AMMO = false;
322         if( mAmmo === 0 )                       def.ST_LOW_AMMO = false;
323       }
324     }

330  D%=[B*ACO/100+R]*(AHP/10)*[(200-(DCO+DTR*DHP))/100]
331 
332  D = Actual damage expressed as a percentage
333 
334  B = Base damage (in damage chart)
335  ACO = Attacking CO attack value (example:130 for Kanbei)
336 
337  R = Random number 0-9
338 
339  AHP = HP of attacker
340 
341  DCO = Defending CO defense value (example:80 for Grimm)
342  DTR = Defending Terrain Stars (IE plain = 1, wood = 2)
343  DHP = HP of defender
344 
345  Denoted as xxxXXXX where x = small star and X = big star..for now.
346  Every star is worth 9000 fund at the start of the game. Each additional
347  use of CO Power(including SCOP) increase the value of each star by 1800
348  fund up to the tenth usage, when it won't increase any further.
349 
350  Stars on your Charge Meter can be charged in two ways:
351 
352  Damaging your oppoent's units. You gain meter equal to half the fund damage
353  you deal. Receiving damage from your opponent. You gain meter equal to the
354  fund damage you take. Keep in mind that AWBW only keeps track of real numbers
355  for the purpose of Charge Meter calculation. That means taking a 57% attack
356  and ending up with 5 hp only adds 0.5*full cost of unit to your Charge Meter.
357  In Summary, the amount of charge added to your meter can be calculated as:
358 
359  (0.5*0.1*Damage Dealt*Cost of Unit X)+(0.1*Damage Received*Cost of Unit Y)
360 
361 
362  It should be noted that a COs meter does not charge during the turn they activate a power.
363 
364  *****************/
