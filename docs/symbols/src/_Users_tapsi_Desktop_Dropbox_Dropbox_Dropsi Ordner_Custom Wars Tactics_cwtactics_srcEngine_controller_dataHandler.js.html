1 /**
  2  *
  3  * Every loader must be configured to load a map from a given map specification
  4  * into the actual model. If an outdated map specification us used, the model
  5  * must be prepared to work with the current model.
  6  * 
  7  * @namespace
  8  */
  9 controller.serializationHandler = {
 10 
 11   /**
 12    * API
 13    * 
 14    * @namespace
 15    */
 16   "interface":{
 17 
 18     /**
 19      * Saves the domain model.
 20      */
 21     save: null,
 22 
 23     /**
 24      * Loads a model into the domain model.
 25      *
 26      * @param {object} data
 27      */
 28     load: null,
 29 
 30     /**
 31      * Serializes an unit object.
 32      *
 33      * @param {object} unit
 34      */
 35     serializeUnit: null,
 36 
 37     /**
 38      * Deserializes an unit object and prepares the model.
 39      *
 40      * @param {Array} data
 41      */
 42     deserializeUnit: null,
 43 
 44     /**
 45      * Serializes an player object.
 46      *
 47      * @param {object} player
 48      */
 49     serializePlayer: null,
 50 
 51     /**
 52      * Deserializes an player object and prepares the model.
 53      *
 54      * @param {Array} data
 55      */
 56     deserializePlayer: null,
 57 
 58     /**
 59      * Serializes a property object.
 60      *
 61      * @param {object} property
 62      */
 63     serializeProperty: null,
 64 
 65     /**
 66      * Deserializes an property object and prepares the model.
 67      *
 68      * @param {Array} data
 69      */
 70     deserializeProperty: null
 71   },
 72 
 73   // #########################################################################
 74 
 75   /**
 76    * Map serialization transfer functions for the milestone 2.6
 77    * map specifications.
 78    * 
 79    * @namespace
 80    */
 81   "2.6":{
 82 
 83     save: function(){
 84       var dom = {};
 85 
 86       // META DATA
 87       dom.day = model.day;
 88       dom.turnOwner = model.turnOwner;
 89       dom.mapWidth = model.mapWidth;
 90       dom.mapHeight = model.mapHeight;
 91 
 92       // MAP
 93       dom.map = [];
 94       var mostIdsMap = {};
 95       var mostIdsMapCurIndex = 0;
 96       for( var x=0,xe=model.mapWidth; x<xe; x++ ){
 97 
 98         dom.map[x] = [];
 99         for( var y=0,ye=model.mapHeight; y<ye; y++ ){
100 
101           var type = dom.map[x][y];
102 
103           if( !mostIdsMap.hasOwnProperty(type) ){
104             mostIdsMap[type] = mostIdsMapCurIndex;
105             mostIdsMapCurIndex++;
106           }
107 
108           dom.map[x][y] = mostIdsMap[type];
109         }
110       }
111 
112       // ADD TYPE MAP
113       dom.typeMap = [];
114       var typeKeys = Object.keys( mostIdsMap );
115       for( var i=0,e=typeKeys.length; i<e; i++ ){
116         dom.typeMap[ mostIdsMap[typeKeys[i]] ] = typeKeys[i];
117       }
118 
119       // UNITS
120       dom.units = [];
121       for( var i=0,e=model.units.length; i<e; i++ ){
122         if( model.units[i].owner !== CWT_INACTIVE_ID ){
123           dom.units.push( this.serializeUnit(model.units[i]) );
124         }
125       }
126 
127       // PROPERTIES
128       dom.properties = [];
129       for( var i=0,e=model.properties.length; i<e; i++ ){
130         if( model.properties[i].owner !== CWT_INACTIVE_ID ){
131           dom.properties.push( this.serializeProperty(model.properties[i]) );
132         }
133       }
134 
135       // PLAYERS
136       dom.players = [];
137       for( var i=0,e=model.players.length; i<e; i++ ){
138         if( model.players[i].team !== CWT_INACTIVE_ID ){
139           dom.players.push( this.serializePlayer(model.players[i]) );
140         }
141       }
142 
143       // ACTORS
144       dom.actors = [];
145       for( var i=0,e=model.leftActors.length; i<e; i++ ){
146         if( model.leftActors[i] ){
147           dom.actors.push( i );
148         }
149       }
150 
151       dom.rules = {};
152       var keys = Object.keys( model.rules );
153       for( var i= 0,e=keys.length; i<e; i++ ){
154         var key = keys[i];
155         dom.rules[ key ] = model.rules[ key ];
156       }
157 
158       return dom;
159     },
160 
161     load: function( data ){
162 
163       model.day = data.day;
164       model.turnOwner = data.turnOwner;
165       model.mapWidth = data.mapWidth;
166       model.mapHeight = data.mapHeight;
167 
168       // MAP
169       for( var x=0,xe=model.mapWidth; x<xe; x++ ){
170         for( var y=0,ye=model.mapHeight; y<ye; y++ ){
171           model.unitPosMap[x][y] = null;
172           model.propertyPosMap[x][y] = null;
173           model.map[x][y] = data.typeMap[ data.map[x][y] ];
174         }
175       }
176 
177       // UNITS
178       for( var i=0,e=model.units.length; i<e; i++ ){
179         model.units[i].owner = CWT_INACTIVE_ID;
180       }
181 
182       for( var i=0,e=data.units.length; i<e; i++ ){
183         this.deserializeUnit( data.units[i] );
184       }
185 
186       // PROPERTIES
187       for( var i=0,e=model.properties.length; i<e; i++ ){
188         model.properties[i].owner = CWT_INACTIVE_ID;
189       }
190 
191       for( var i=0,e=data.properties.length; i<e; i++ ){
192         this.deserializeProperty( data.properties[i] );
193       }
194 
195       // PLAYERS
196       for( var i=0,e=model.players.length; i<e; i++ ){
197         model.players[i].team = CWT_INACTIVE_ID;
198       }
199 
200       for( var i=0,e=data.players.length; i<e; i++ ){
201         this.deserializePlayer( data.players[i] );
202       }
203 
204       // ACTORS
205       for( var i=0,e=model.leftActors.length; i<e; i++ ){
206         model.leftActors[i] = false;
207       }
208 
209       for( var i=0,e=data.leftActors.length; i<e; i++ ){
210         model.leftActors[ data.leftActors[i] ] = true;
211       }
212 
213       model.setRulesByOption( data.rules );
214 
215     },
216 
217     serializeUnit: function( unit ){
218 
219       return [
220         model.extractUnitId(unit),
221         unit.type,
222         unit.x,
223         unit.y,
224         unit.hp,
225         unit.ammo,
226         unit.fuel,
227         unit.loadedIn,
228         unit.owner
229       ];
230     },
231 
232     deserializeUnit: function( data ){
233 
234       // GET UNIT
235       var id = data[0];
236       var unit = model.units[id];
237 
238       // INJECT DATA
239       unit.type     = data[1];
240       unit.x        = data[2];
241       unit.y        = data[3];
242       unit.hp       = data[4];
243       unit.ammo     = data[5];
244       unit.fuel     = data[6];
245       unit.loadedIn = data[7];
246       unit.owner    = data[8];
247 
248       model.unitPosMap[ data[2] ][ data[3] ] = unit;
249     },
250 
251     serializePlayer: function( player ){
252 
253       return [
254         player.extractPlayerId( player ),
255         player.name,
256         player.gold,
257         player.team
258       ];
259     },
260 
261     deserializePlayer: function( data ){
262 
263       // GET PLAYER
264       var id = data[0];
265       var player = model.players[id];
266 
267       // INJECT DATA
268       player.name = data[1];
269       player.gold = data[2];
270       player.team = data[3];
271     },
272 
273     serializeProperty: function( property ){
274 
275       // SEARCH POSITION
276       var px,py;
277       var found = false;
278       for( var x=0; x<model.mapWidth && !found; x++ ){
279         for( var y=0; y<model.mapHeight && !found; y++ ){
280           if( model.propertyPosMap[x][y] === property ){
281             px = x;
282             py = y;
283             found = true;
284           }
285         }
286       }
287 
288       return [
289         model.extractPropertyId( property ),
290         px,
291         py,
292         property.type,
293         property.capturePoints,
294         property.owner
295       ];
296     },
297 
298     deserializeProperty: function( data ){
299 
300       // GET PROPERTY
301       var id = data[0];
302       var property = model.properties[id];
303 
304       // INJECT DATA
305       property.type          = data[3];
306       property.capturePoints = data[4];
307       property.owner         = data[5];
308 
309       model.propertyPosMap[ data[1] ][ data[2] ] = property;
310     }
311   }
312  
313 };
314 
315 /**
316  * Current active serialization handler version.
317  * 
318  * @constant
319  */
320 controller.CURRENT_SERIALIZATION_HANDLER = "2.6";
321 
322 /**
323  * Returns the correct {@link controller.serializationHandler} object for the given 
324  * version.
325  * 
326  * @param {String} version 
327  * @returns serialization handler object
328  */
329 controller.getActiveSerializationHandler = function( version ){
330   if( arguments.length === 1 ){
331     if( DEBUG &&  !controller.serializationHandler.hasOwnProperty( version ) ){
332       util.raiseError("unknown map format");
333     }
334 
335     return controller.serializationHandler[ version ];
336   }
337   else return controller.serializationHandler[ controller.CURRENT_SERIALIZATION_HANDLER ];
338 };
