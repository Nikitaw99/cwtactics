 94 /**
 95  * Returns the html image object for the given unit type in the given color
 96  * shema and state.
 97  *
 98  * @param type
 99  * @param code
100  * @param color
101  */
102 images.getUnitImageForType = function( type, code, color ){
103 
104   switch( code ){
105 
106     case images.IMAGE_CODE_IDLE :
107       return images.CodeIdleImages[color][type];
108 
109     case images.IMAGE_CODE_IDLE_INVERTED :
110       return images.CodeIdleInvertedImages[color][type];
111 
112     case images.IMAGE_CODE_LEFT :
113       return images.CodeLeftImages[color][type];
114 
115     case images.IMAGE_CODE_RIGHT :
116       return images.CodeRightImages[color][type];
117 
118     case images.IMAGE_CODE_DOWN :
119       return images.CodeDownImages[color][type];
120 
121     case images.IMAGE_CODE_UP :
122       return images.CodeUpImages[color][type];
123 
124     case images.IMAGE_CODE_STATELESS :
125       return images.CodeStatelessImages[color][type];
126 
127     default: util.logError("unknown image state code ",code );
128   }
129 };

154 /**
155  * Loads all images on client initialization.
156  */
157 signal.connect( signal.EVENT_CLIENT_INIT, function(){
158   var _loadingImages = [];
159 
160   function imageToLoader( type, code, color ){
161     if( util.DEBUG ){
162       util.logInfo("loading images for",type,"code",code,"color",color);
163     }
164 
165     var img = new Image();
166     img.src = images.IMAGE_PATH+color+"_"+code+"_"+type+".png";
167     _loadingImages.push( img, type, code, color );
168   }
169 
170   function propertyImageToLoader( type, color ){
171     if( util.DEBUG ){
172       util.logInfo("loading images for",type,"color",color);
173     }
174 
175     var img = new Image();
176     img.src = images.IMAGE_PATH+"PROP_"+color+"_"+type+".png";
177     _loadingImages.push( img, type, images.IMAGE_CODE_STATELESS, color );
178   }
179 
180   function tileImageToLoader( type ){
181     if( util.DEBUG ){
182       util.logInfo("loading images for",type);
183     }
184 
185     var img = new Image();
186     img.src = images.IMAGE_PATH+type+".png";
187     _loadingImages.push( img, type, images.IMAGE_CODE_STATELESS,
188            images.COLOR_NONE );
189   }
190 
191   function unitTypeToLoader( type, color ){
192     imageToLoader(type, images.IMAGE_CODE_IDLE, color );
193     imageToLoader(type, images.IMAGE_CODE_IDLE_INVERTED, color );
194     imageToLoader(type, images.IMAGE_CODE_LEFT, color );
195     imageToLoader(type, images.IMAGE_CODE_RIGHT, color );
196     imageToLoader(type, images.IMAGE_CODE_UP, color );
197     imageToLoader(type, images.IMAGE_CODE_DOWN, color );
198   }
199 
200   var units = game.getListOfUnitTypes();
201   for( var i=0,e=units.length; i<e; i++ ){
202 
203     var type = units[i];
204     unitTypeToLoader( type, images.COLOR_RED );
205     unitTypeToLoader( type, images.COLOR_GREEN );
206     unitTypeToLoader( type, images.COLOR_BLUE );
207     unitTypeToLoader( type, images.COLOR_BLACK_MASK );
208   }
209 
210   var props = game.getListOfPropertyTypes();
211   for( var i=0,e=props.length; i<e; i++ ){
212 
213     var type = props[i];
214     propertyImageToLoader( type, images.COLOR_RED );
215     propertyImageToLoader( type, images.COLOR_GREEN );
216     propertyImageToLoader( type, images.COLOR_BLUE );
217     propertyImageToLoader( type, images.COLOR_NEUTRAL );
218 
219     if( util.DEBUG ){
220       util.logInfo("loading images for",type);
221     }
222   }
223 
224   var tiles = game.getListOfTileTypes();
225   for( var i=0,e=tiles.length; i<e; i++ ){
226     tileImageToLoader( tiles[i] );
227     if( util.DEBUG ){
228       util.logInfo("loading image for",tiles[i]);
229     }
230   }
231 
232   tileImageToLoader( "HP_1" );
233   tileImageToLoader( "HP_2" );
234   tileImageToLoader( "HP_3" );
235   tileImageToLoader( "HP_4" );

263   var checkImageLoadingStatus = function(){
264 
265     // CHECK THEM
266     for( var i=0,e=_loadingImages.length; i<e; i+=4 ){
267 
268       // NOT ALL IMAGES ARE LOADED BREAK
269       if( _loadingImages[i].complete !== true ){
270         return false;
271       }
272     }
273 
274     if( util.DEBUG ){
275       util.logInfo("images loaded, adding them to client layer");
276     }
277 
278     // ALL IMAGES ARE LOADED, ADD THEM TO IMAGE DATA BASE
279     for( var i=0,e=_loadingImages.length; i<e; i+=4 ){
280 
281       var type  = _loadingImages[i+1];
282       var code  = _loadingImages[i+2];
283       var color = _loadingImages[i+3];
284       switch( code ){
285 
286 
287         case images.IMAGE_CODE_IDLE :
288           images.CodeIdleImages[color][type]         = _loadingImages[i];
289           break;
290 
291         case images.IMAGE_CODE_STATELESS :
292           images.CodeStatelessImages[color][type]    = _loadingImages[i];
293           break;
294 
295         case images.IMAGE_CODE_IDLE_INVERTED :
296           images.CodeIdleInvertedImages[color][type] = _loadingImages[i];
297           break;
298 
299         case images.IMAGE_CODE_LEFT :
300           images.CodeLeftImages[color][type]         = _loadingImages[i];
301           break;
302 
303         case images.IMAGE_CODE_RIGHT :
304           images.CodeRightImages[color][type]        = _loadingImages[i];
305           break;
306 
307         case images.IMAGE_CODE_DOWN :
308           images.CodeDownImages[color][type]         = _loadingImages[i];
309           break;
310 
311         case images.IMAGE_CODE_UP :
312           images.CodeUpImages[color][type]           = _loadingImages[i];
313           break;
314 
315         default: util.logError("unknown image state code ",code );
316       }
317     }
318 
319     // remove helper
320     _loadingImages.splice(0);
321 
322     if( util.DEBUG ) util.logInfo("images successfully loaded");
323 
324     return true;
325   };
326 
327   function loadIt(){
328     if(!checkImageLoadingStatus() ) setTimeout( loadIt, 100 );
329     else startClient();
330   }
331   loadIt();
332 });
