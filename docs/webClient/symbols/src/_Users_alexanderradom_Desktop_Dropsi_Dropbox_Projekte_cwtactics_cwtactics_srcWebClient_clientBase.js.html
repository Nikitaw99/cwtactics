 13 /**
 14  * Updates the game logic and graphic.
 15  *
 16  * @param delta time since last call in ms
 17  */
 18 client.update = function( delta ){
 19 
 20   if( animation.isMoveAnimationActive() === false ){
 21 
 22     // UPDATE LOGIC
 23     if( controller.hasNextAction() ){
 24       var msg = controller.evaluateNextAction();
 25 
 26       if( msg.k === "move" ){
 27         animation.prepareMoveAnimation(msg.a[0], msg.a[1],msg.a[2], msg.a[3]);
 28       }
 29 
 30       if( msg.k === "silofire" ){
 31         var x = msg.a[5];
 32         var y = msg.a[6];
 33         screen.markForRedrawRange( x,y,2 );
 34       }
 35 
 36       if( util.DEBUG ){
 37         util.logInfo("Executed a",msg.k,"command");
 38       }
 39     }
 40   }
 41   else{
 42 
 43     // UPDATE MOVE
 44     animation.updateMoveAnimation( delta );
 45   }
 46 
 47   // UPDATE ANIMATION STEP
 48   if( userInput.moveScreenX === 0 &&
 49       userInput.moveScreenY === 0 ){
 50 
 51     animation.updateAnimationTimer( delta );
 52   }
 53   else{
 54     userInput.solveMapShift();
 55   }
 56 
 57   // RENDER
 58   if( screen.drawChanges > 0 ){
 59     screen.drawScreen();
 60   }
 61 
 62   // RENDER MOVE
 63   if( animation.isMoveAnimationActive() === true ){
 64 
 65     // uid, cx, cy, shift, moveCode
 66     screen.drawMoveAnimation(
 67       animation.moveAnimationUid,
 68       animation.moveAnimationX, animation.moveAnimationY,
 69       animation.moveAnimationShift,
 70       animation.moveAnimationPath[ animation.moveAnimationIndex ]
 71     );
 72   }
 73 };
