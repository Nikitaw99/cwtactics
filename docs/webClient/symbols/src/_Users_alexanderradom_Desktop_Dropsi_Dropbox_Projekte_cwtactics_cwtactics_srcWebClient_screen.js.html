 89 screen._drawTile = function( x,y,tx,ty ){
 90   var canvasCtx = screen._canvasCtx;
 91   var  tp = game.tileByPos(x,y);
 92   var pic = images.getTileImageForType( tp );
 93 
 94   var scx = 0;
 95   var scy = 0;
 96   var scw = 32;
 97   var sch = 64;
 98   var tcx = (x- screen.screenX )*tx;
 99   var tcy = (y- screen.screenY )*ty - ty;
100   var tcw = tx;
101   var tch = ty+ty;
102 
103   if( tcy < 0 ){
104     scy = scy + ty;
105     sch = sch - ty;
106     tcy = tcy + ty;
107     tch = tch - ty;
108   }
109 
110   if( pic !== undefined ){
111     canvasCtx.drawImage(
112       pic,
113       scx,scy,
114       scw,sch,
115       tcx,tcy,
116       tcw,tch
117     );
118   }
119   else{
120     canvasCtx.fillStyle="rgb(0,0,255)";
121     canvasCtx.fillRect( tcx,tcy, tcw,tch );
122   }
123 };

236 screen._drawUnit = function( x,y,tx,ty,step ){
237   var canvasCtx = screen._canvasCtx;
238 
239   // draw unit above tile
240   var unit = game.unitByPos(x,y);
241   if( unit !== null ){
242     if( unit === screen.preventRenderUnit ) return;
243 
244     var color;
245     if( unit.owner === game.getTurnOwner() ){
246       color = images.COLOR_GREEN;
247     }
248     else if( game.player(unit.owner).team ===
249       game.player( game.getTurnOwner() ).team ){
250 
251       color = images.COLOR_BLUE;
252     }
253     else color = images.COLOR_RED;
254 
255     var state = ( unit.owner % 2 === 1 )?
256       images.IMAGE_CODE_IDLE : images.IMAGE_CODE_IDLE_INVERTED;
257 
258     var pic = images.getUnitImageForType( unit.type, state, color );
259 
260     var scx = 64* animation.unitAnimationStep();
261     var scy = 0;
262     var scw = 64;
263     var sch = 64;
264     var tcx = (x- screen.screenX )*tx -16;
265     var tcy = (y- screen.screenY )*ty -16;
266     var tcw = tx+tx;
267     var tch = ty+ty;
268 
269     if( pic !== undefined ){
270 
271       canvasCtx.drawImage(
272         pic,
273         scx,scy,
274         scw,sch,
275         tcx,tcy,
276         tcw,tch
277       );
278 
279       // RENDER GRAY OVERLAY TO MARK AS USED
280       if( unit.owner === game.getTurnOwner() &&
281         !game.canAct( game.extractUnitId( unit ) ) ){
282 
283         canvasCtx.globalAlpha = 0.15;
284         canvasCtx.drawImage(
285           images.getUnitImageForType(
286             unit.type, state, images.COLOR_BLACK_MASK
287           ),
288           scx,scy,
289           scw,sch,
290           tcx,tcy,
291           tcw,tch
292         );
293         canvasCtx.globalAlpha = 1;
294       }
295     }
296     else{
297 
298       canvasCtx.fillStyle="rgb(255,0,0)";
299       canvasCtx.fillRect( tcx+8,tcy+8, tcw/2,tch/2 );
300     }
301 
302     if( unit.hp <= 90 ){
303 
304       var pic = undefined;
305            if( unit.hp > 80 ) pic = images.getTileImageForType("HP_9");
306       else if( unit.hp > 70 ) pic = images.getTileImageForType("HP_8");
307       else if( unit.hp > 60 ) pic = images.getTileImageForType("HP_7");
308       else if( unit.hp > 50 ) pic = images.getTileImageForType("HP_6");
309       else if( unit.hp > 40 ) pic = images.getTileImageForType("HP_5");
310       else if( unit.hp > 30 ) pic = images.getTileImageForType("HP_4");
311       else if( unit.hp > 20 ) pic = images.getTileImageForType("HP_3");
312       else if( unit.hp > 10 ) pic = images.getTileImageForType("HP_2");
313       else pic = images.getTileImageForType("HP_1");
314 
315       canvasCtx.drawImage(
316         pic,
317         tcx+32,tcy+32,
318         16,16
319       );
320     }
321 
322     if( step <= 2 &&
323       screen.ammoStatusMap[x- screen.screenX][y- screen.screenY] === true ){
324 
325       pic = images.getTileImageForType("UNIT_STAT_AMMO");
326       canvasCtx.drawImage(
327         pic,
328         tcx+16,tcy+32,
329         16,16
330       );
331     }
332 
333     if( step >= 4 && step <= 6 &&
334       screen.fuelStatusMap[x- screen.screenX][y- screen.screenY] === true){
335 
336       pic = images.getTileImageForType("UNIT_STAT_FUEL");
337       canvasCtx.drawImage(
338         pic,
339         tcx+16,tcy+32,
340         16,16
341       );
342     }
343   }
344 };

346 /**
347  * Draws a arrow for the active move path on the screen.
348  *
349  * @param ctx
350  * @private
351  */
352 screen._drawMoveArrow = function(){
353   var ctx = screen._canvasCtx;
354   var tileX = screen.tileSizeX;
355   var tileY = screen.tileSizeY;
356   var currentMovePath = userInput.currentMovePath;
357   var cX = userInput.currentMovePathSourceX;
358   var cY = userInput.currentMovePathSourceY;
359   var oX;
360   var oY;
361   var tX;
362   var tY;
363   var pic;
364 
365   for( var i=0,e=currentMovePath.length; i<e; i++ ){
366     if( currentMovePath[i] === null ) break;
367 
368     var oX = cX;
369     var oY = cY;
370 
371     // CURRENT TILE
372     switch( currentMovePath[i] ){
373       case game.MOVE_CODE_UP :    cY--; break;
374       case game.MOVE_CODE_RIGHT : cX++; break;
375       case game.MOVE_CODE_DOWN :  cY++; break;
376       case game.MOVE_CODE_LEFT :  cX--; break;
377     }
378 
379     // NEXT TILE
380     if( i === e-1 || currentMovePath[i+1] === null ){
381       tX = -1; tY = -1;
382     }
383     else{
384       switch( currentMovePath[i+1] ){
385         case game.MOVE_CODE_UP :    tX = cX;   tY = cY-1; break;
386         case game.MOVE_CODE_RIGHT : tX = cX+1; tY = cY;   break;
387         case game.MOVE_CODE_DOWN :  tX = cX;   tY = cY+1; break;
388         case game.MOVE_CODE_LEFT :  tX = cX-1; tY = cY;   break;
389       }
390     }
391 
392     if( tX == -1 ){
393 
394       // TARGET TILE
395       switch( currentMovePath[i] ){
396         case game.MOVE_CODE_UP :
397           pic = images.getTileImageForType("ARROW_T_N"); break;
398         case game.MOVE_CODE_RIGHT :
399           pic = images.getTileImageForType("ARROW_T_E"); break;
400         case game.MOVE_CODE_DOWN :
401           pic = images.getTileImageForType("ARROW_T_S"); break;
402         case game.MOVE_CODE_LEFT :
403           pic = images.getTileImageForType("ARROW_T_W"); break;
404       }
405     }
406     else{
407 
408       var diffX = Math.abs( tX-oX );
409       var diffY = Math.abs( tY-oY );
410 
411       // IN THE MIDDLE OF THE WAY
412       if( diffX === 2 ){
413         pic = images.getTileImageForType("ARROW_L_EW");
414       }
415       else if( diffY === 2 ){
416         pic = images.getTileImageForType("ARROW_L_NS");
417       }
418       else if( (tX<cX && oY>cY) || (oX<cX && tY>cY)  ){
419         pic = images.getTileImageForType("ARROW_E_SW");
420       }
421       else if( (tX<cX && oY<cY) || (oX<cX && tY<cY) ){
422         pic = images.getTileImageForType("ARROW_E_WN");
423       }
424       else if( (tX>cX && oY<cY) || (oX>cX && tY<cY) ){
425         pic = images.getTileImageForType("ARROW_E_NE");
426       }
427       else if( (tX>cX && oY>cY) || (oX>cX && tY>cY) ){
428         pic = images.getTileImageForType("ARROW_E_ES");
429       }
430       else{
431         util.logError(
432           "illegal move arrow state",
433           "old (",oX,",",oY,")",
434           "current (",cX,",",cY,")",
435           "next (",tX,",",tY,")",
436           "path (", currentMovePath ,")"
437         );
438 
439         continue;
440       }
441     }
442 
443     cX = cX - screen.screenX;
444     cY = cY - screen.screenY;
445     if( cX >= 0 && cY >= 0 &&
446       cX < screen.screenWidth && cY < screen.screenHeight ){
447 
448       ctx.drawImage( pic,cX*tileX,cY*tileY );
449     }
450   }
451 };

453 /**
454  * Draws a moving unit.
455  *
456  * @param uid mover id
457  * @param cx current y position
458  * @param cy current x position
459  * @param shift shift thats already moved on the current tile
460  * @param moveCode move code of the current position in the move path
461  */
462 screen.drawMoveAnimation = function( uid, cx, cy, shift, moveCode ){
463   var unit = game.unitById( uid );
464 
465   var color;
466   if( unit.owner === game.getTurnOwner() ){
467     color = images.COLOR_GREEN;
468   }
469   else if( game.player(unit.owner).team ===
470     game.player( game.getTurnOwner() ).team ){
471 
472     color = images.COLOR_BLUE;
473   }
474   else color = images.COLOR_RED;
475 
476   var state;
477   var tp = unit.type;
478 
479   // GET CORRECT IMAGE STATE
480   switch( moveCode ){
481     case game.MOVE_CODE_UP :    state = images.IMAGE_CODE_UP;    break;
482     case game.MOVE_CODE_RIGHT : state = images.IMAGE_CODE_RIGHT; break;
483     case game.MOVE_CODE_DOWN :  state = images.IMAGE_CODE_DOWN;  break;
484     case game.MOVE_CODE_LEFT :  state = images.IMAGE_CODE_LEFT;  break;
485   }
486 
487   var pic = images.getUnitImageForType( unit.type, state, color );
488 
489   var scx = 64* animation.unitAnimationStep();
490   var scy = 0;
491   var scw = 64;
492   var sch = 64;
493   var tcx = ( cx- screen.screenX )*screen.tileSizeY -16;
494   var tcy = ( cy- screen.screenY )*screen.tileSizeY -16;
495   var tcw = screen.tileSizeX+screen.tileSizeX;
496   var tch = screen.tileSizeY+screen.tileSizeY;
497 
498   // ADD SHIFT
499   switch( moveCode ){
500     case game.MOVE_CODE_UP:    tcy -= shift; break;
501     case game.MOVE_CODE_LEFT:  tcx -= shift; break;
502     case game.MOVE_CODE_RIGHT: tcx += shift; break;
503     case game.MOVE_CODE_DOWN:  tcy += shift; break;
504   }
505 
506   // DRAW IT
507   screen._canvasCtx.drawImage(
508     pic,
509     scx,scy,
510     scw,sch,
511     tcx,tcy,
512     tcw,tch
513   );
514 };

516 /**
517  * Draws the screen.
518  */
519 screen.drawScreen = function(){
520 
521   var xe,ye;
522   var screen = window.screen;
523   var tx = screen.tileSizeX;
524   var ty = screen.tileSizeY;
525   var focusExists = (
526     controller.currentState === controller.STATE_SELECT_MOVE_PATH ||
527       controller.currentState === controller.STATE_SELECT_ACTION_TARGET
528     );
529 
530   // iterate by row
531   ye = screen.screenY + screen.screenHeight -1;
532   if( ye >= game.mapHeight() ) ye = game.mapHeight()-1;
533   for(var y=screen.screenY; y<=ye; y++){
534 
535     // iterate by column
536     xe = screen.screenX + screen.screenWidth -1;
537     if( xe >= game.mapWidth() ) xe = game.mapWidth()-1;
538     for(var x= screen.screenX; x<=xe; x++){
539 
540       // RENDER IF NEEDED
541       if( screen.drawnMap[ x- screen.screenX ][ y- screen.screenY ] > 0 ){
542 
543         screen._drawTile(     x, y, tx, ty );
544         screen._drawProperty( x, y, tx, ty );
545         if( focusExists ) screen._drawFocusTile(  x, y, tx, ty );
546         screen._drawUnit(     x, y, tx, ty , animation.statAnimationStep() );
547 
548         screen.drawnMap[ x- screen.screenX ][ y- screen.screenY ] = 0;
549       }
550     }
551   }
552 
553   // DRAW ARROW
554   if( controller.currentState === controller.STATE_SELECT_MOVE_PATH ){
555     screen._drawMoveArrow();
556   }
557 
558   // clear counter
559   screen.drawChanges = 0;
560 };

576 /**
577  * Marks a position for re-rendering.
578  *
579  * @param x
580  * @param y
581  */
582 screen.markForRedraw = function( x,y ){
583   var rX = x;
584   var rY = y;
585   if( x >= 0 && y >= 0 && x < game.mapWidth() && y < game.mapHeight() ){
586     x = x - screen.screenX;
587     y = y - screen.screenY;
588     if( x >= 0 && y >= 0 &&
589       x < screen.screenWidth && y < screen.screenHeight ){
590 
591       if( screen.drawnMap[x][y] === 1 ) return;
592 
593       screen.drawnMap[x][y] = 1;
594       screen.drawChanges++;
595 
596       // check bottom tile
597       x = rX;
598       y = rY+1;
599       if( y < game.mapHeight() ){
600         if( game.propertyByPos(x,y) !== null ) screen.markForRedraw(x,y);
601         else{
602           var tp = game.tileByPos(x,y);
603           if( screen.OVERLAYER[tp] === true ) screen.markForRedraw(x,y);
604         }
605       }
606     }
607   }
608   else util.logError("illegal arguments ",x,",",y," -> out of screen bounds");
609 };

611 /**
612  * Rerenders a tile and all neightbours in a range around it.
613  *
614  * @example
615  *
616  * r = 2;
617  *
618  *      x
619  *    x x x
620  *  x x o x x
621  *    x x x
622  *      x
623  *
624  * @param x
625  * @param y
626  */
627 screen.markForRedrawRange = function( x,y,r ){
628   var lX;
629   var hX;
630   var lY = y-r;
631   var hY = y+r;
632   if( lY < 0 ) lY = 0;
633   if( hY >= domain.mapHeight ) hY = domain.mapHeight-1;
634   for( ; lY<=hY; lY++ ){
635 
636     var disY = Math.abs( lY-y );
637     lX = x-r+disY;
638     hX = x+r-disY;
639     if( lX < 0 ) lX = 0;
640     if( hX >= domain.mapWidth ) hX = domain.mapWidth-1;
641     for( ; lX<=hX; lX++ ){
642 
643       screen.markForRedraw(lX,lY);
644     }
645   }
646 };

648 /**
649  * Rerenders a tile and all 4 neightbours in a plus around it.
650  *
651  * @example
652  *    x
653  *  x o x
654  *    x
655  *
656  * @param x
657  * @param y
658  */
659 screen.markForRedrawWithNeighbours = function( x,y ){
660 
661   if( y>0 ) screen.markForRedraw(x,y-1);
662   if( x>0 ) screen.markForRedraw(x-1,y);
663   screen.markForRedraw(x,y);
664   if( y< game.mapHeight()-1 ) screen.markForRedraw(x,y+1);
665   if( x< game.mapWidth()-1 )  screen.markForRedraw(x+1,y);
666 };

668 /**
669  * Rerenders a tile and all 8 neightbours around it.
670  *
671  * @example
672  *  x x x
673  *  x o x
674  *  x x x
675  *
676  * @param x
677  * @param y
678  */
679 screen.markForRedrawWithNeighboursRing = function( x,y ){
680   var gW = game.mapWidth();
681   var gH = game.mapHeight();
682 
683   // LEFT COLUMN
684   if( x>0 ){
685     if( y>0 )     screen.markForRedraw(x-1,y-1);
686                   screen.markForRedraw(x-1,y);
687     if( y< gH-1 ) screen.markForRedraw(x-1,y+1);
688   }
689 
690   // MIDDLE COLUMN
691   if( y>0 )       screen.markForRedraw(x,y-1);
692                   screen.markForRedraw(x,y);
693   if( y< gH-1 )   screen.markForRedraw(x,y+1);
694 
695   // RIGHT COLUMN
696   if( x< gW-1 ){
697     if( y>0 )     screen.markForRedraw(x+1,y-1);
698                   screen.markForRedraw(x+1,y);
699     if( y< gH-1 ) screen.markForRedraw(x+1,y+1);
700   }
701 };
