  5 sound.play = function( key ){
  6   if( sound._enabled !== true ) return;
  7 
  8   var buffer = sound._sounds[key];
  9   var source = sound.context.createBufferSource();
 10   source.buffer = buffer;
 11   source.connect( sound.context.destination );
 12   source.noteOn(0);
 13 };
 14 
 15 sound._enabled = false;
 16 
 17 sound.enable = function(){
 18   if( sound._enabled === false ){
 19     sound._enabled = true;

 24 sound.loadSound = function( key, url ){
 25   var request = new XMLHttpRequest();
 26   request.open('GET', url, true);
 27   request.responseType = 'arraybuffer';
 28 
 29   // Decode asynchronously
 30   request.onload = function(){
 31     sound.context.decodeAudioData( request.response, function(buffer) {
 32       sound._sounds[key] = buffer;
 33 
 34 //      sound.playBG("LASH");
 35     }, null );
 36   };
 37 
 38   util.logInfo("try to load sound");
 39   try{
 40     request.send();
 41   }
 42   catch(e){
 43     util.logError("could not load sound");
 44     sound._enabled = -1; // NEVER ENABLE
 45   }
 46 };
