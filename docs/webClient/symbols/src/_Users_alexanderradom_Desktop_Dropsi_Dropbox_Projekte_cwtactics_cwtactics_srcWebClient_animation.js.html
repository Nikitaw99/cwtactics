1 /**
  2  * Creates a timerable object with holdable step state.
  3  *
  4  * @private
  5  */
  6 animation._createTimer = function( steps, timePerStep ){
  7   var cTime = 0;
  8   var cStep = 0;
  9 
 10   return {
 11     update: function( delta ){
 12       cTime += delta;
 13       if( cTime >= timePerStep ){
 14         // INCREASE STEP AND RESET TIMER
 15         cTime = 0;
 16         cStep++;
 17         if( cStep >= steps ) cStep = 0;
 18         return true;
 19       }
 20       return false;
 21     },
 22 
 23     currentStep : function(){
 24       return cStep;
 25     }
 26   }
 27 }

 79 /**
 80  * Updates the animation timer step.
 81  *
 82  * @param delta delta time in milliseconds
 83  */
 84 animation.updateAnimationTimer = function( delta ){
 85 
 86   var uni = animation.unitAnimation.update( delta );
 87   var pro = animation.propertyAnimation.update( delta );
 88   var sel = animation.selectorAnimation.update( delta );
 89   animation.statAnimation.update( delta );
 90 
 91   var x  = screen.screenX;
 92   var yS = screen.screenY;
 93   var xe = x+ screen.screenWidth;
 94   var ye = yS+screen.screenHeight;
 95 
 96   // CHECK BOUNDS
 97   if( xe > game.mapWidth() )  xe = game.mapWidth();
 98   if( ye > game.mapHeight() ) ye = game.mapHeight();
 99 
100   // ITERATE THROUGH THE SCREEN
101   for( ; x<xe; x++ ){
102     for( var y=yS ; y<ye; y++ ){
103 
104       if( uni ){
105         var unit = game.unitByPos(x,y);
106         if( unit !== null ){
107           screen.markForRedrawWithNeighbours(x,y);
108         }
109       }
110 
111       if( pro ){
112         var prop = game.propertyByPos(x,y);
113         if( prop !== null ){
114           if( y>0 ) screen.markForRedraw(x,y-1);
115                     screen.markForRedraw(x,y);
116         }
117       }
118 
119       if( sel ){
120         if( controller.valueOfSelectionPos(x,y) > -1 ){
121           screen.markForRedraw(x,y);
122         }
123       }
124     }
125   }
126 
127 
128 };

158 /**
159  * Prepares a move animation for drawing.
160  */
161 animation.prepareMoveAnimation = function( uid, x,y, path ){
162   var unit = game.unitById( uid );
163 
164   animation.moveAnimationX = x;
165   animation.moveAnimationY = y;
166   animation.moveAnimationIndex = 0;
167   animation.moveAnimationPath = path;
168   animation.moveAnimationUid = uid;
169   animation.moveAnimationShift = 0;
170 
171   if( util.DEBUG ){
172     util.logInfo(
173       "drawing move from",
174       "(",animation.moveAnimationX,",",animation.moveAnimationY,")",
175       "with path",
176       "(",animation.moveAnimationPath,")");
177   }
178 
179   screen.preventRenderUnit = unit;
180 };

189 /**
190  *
191  */
192 animation.updateMoveAnimation = function( delta ){
193 
194   // MOVE 4 TILES / SECOND
195   animation.moveAnimationShift += (delta/1000)*screen.tileSizeX*14;
196 
197   screen.markForRedrawWithNeighboursRing(
198     animation.moveAnimationX, animation.moveAnimationY
199   );
200 
201   if( animation.moveAnimationShift > screen.tileSizeX ){
202 
203     // UPDATE ANIMATION POS
204     switch( animation.moveAnimationPath[ animation.moveAnimationIndex ] ){
205       case game.MOVE_CODE_UP :    animation.moveAnimationY--; break;
206       case game.MOVE_CODE_RIGHT : animation.moveAnimationX++; break;
207       case game.MOVE_CODE_DOWN :  animation.moveAnimationY++; break;
208       case game.MOVE_CODE_LEFT :  animation.moveAnimationX--; break;
209     }
210 
211     animation.moveAnimationIndex++;
212 
213     // moveAnimationShift = 0;
214     animation.moveAnimationShift -= screen.tileSizeX;
215 
216     // MOVE ANIMATION HAS ENDED
217     if( animation.moveAnimationIndex === animation.moveAnimationPath.length ){
218       animation.moveAnimationX = -1;
219       animation.moveAnimationY = -1;
220       animation.moveAnimationIndex = 0;
221       animation.moveAnimationUid = -1;
222       animation.moveAnimationPath = null;
223 
224       screen.preventRenderUnit = null; // RENDER UNIT NOW NORMALLY
225     }
226   }
227 };
