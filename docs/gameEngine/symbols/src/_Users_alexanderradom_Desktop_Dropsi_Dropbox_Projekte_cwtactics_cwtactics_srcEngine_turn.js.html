 12 /**

 26 /**
 27  * Returns true if the selected uid can act in the current active turn,
 28  * else false.
 29  *
 30  * @param uid selected unit identical number
 31  */
 32 game.canAct = function( uid ){
 33   var startIndex = domain.turnOwner * CWT_MAX_UNITS_PER_PLAYER;
 34 
 35   // NOT THE OWNER OF THE CURRENT TURN
 36   if( uid >= startIndex + CWT_MAX_UNITS_PER_PLAYER || uid < startIndex ){
 37 
 38     return false;
 39   }
 40 
 41   var index = uid - startIndex;
 42   return domain.leftActors[ index ] === true;
 43 };

 45 /**
 46  * Returns the team id that won a round by annithilating all enemies or
 47  * -1 if more than one team is alive.
 48  */
 49 game.annithilationWinner = function(){
 50   var team;
 51   for( var i=0; i<CWT_MAX_PLAYER; i++ ){
 52 
 53     var plC = domain.players[i];
 54     if( plC === null ) continue;
 55     var teamC = plC.team;
 56     if( team === undefined ) team = teamC;
 57     else if( team !== teamC ){
 58 
 59       // TWO OPPOSITE TEAMS EXISTS -> NO WINNER
 60       return CWT_INACTIVE_ID;
 61     }
 62 
 63     // ONLY ONE TEAM FOUND -> THIS IS THE WINNER TEAM
 64     return team;
 65   }
 66 };

 82 /**
 83  * Sets the next available player as turn owner. The search algorithm starts
 84  * with the id of the current turn owner +1. If the player which owns this id
 85  * is not alive, another +1 will be added to the searched id. If the last id
 86  * of the player list is also not alive the search will start with 0 again.
 87  * The day counter increases if this situation happens.
 88  */
 89 game.setNextPlayerAsTurnOwner = function(){
 90 
 91   var pid = domain.turnOwner;
 92   var oid = pid;
 93 
 94   pid++;
 95 
 96   // FIND NEXT PLAYER
 97   while( pid !== oid ){
 98 
 99     if( game.player( pid ) !== null ){
100 
101       // FOUND NEXT PLAYER
102       break;
103     }
104 
105     // INCREASE ID
106     pid++;
107     if( pid === CWT_MAX_PLAYER ) pid = 0;
108   }
109 
110   // CHECK NEW
111   if( pid === oid ) util.logError("cannot find next player");
112   domain.turnOwner = pid;
113 
114   var startIndex= pid* CWT_MAX_UNITS_PER_PLAYER;
115   for( var i= startIndex,
116            e= i+CWT_MAX_UNITS_PER_PLAYER; i<e; i++ ){
117 
118     domain.leftActors[i-startIndex] = (
119       game.unitById(i) !== null )? true : false;
120   }
121 };

123 /**
124  * Removes an unit from the actable array. An unit that goes into
125  * the wait status cannot do another action in the active turn.
126  *
127  * @param uid
128  */
129 game.markAsUnusable = function( uid ){
130 
131   var uid = ( typeof uid === 'number' )? uid : game.extractUnitId( uid );
132   var startIndex = domain.turnOwner * CWT_MAX_UNITS_PER_PLAYER;
133 
134   // NOT THE OWNER OF THE CURRENT TURN
135   if( uid >= startIndex + CWT_MAX_UNITS_PER_PLAYER ||
136     uid < startIndex ){
137 
138     util.logError("unit owner is not the active player");
139   }
140 
141   domain.leftActors[ uid - startIndex ] = false;
142 
143   if( util.DEBUG ){
144     util.logInfo("unit",uid,"going into wait status");
145   }
146 };

153 /**
154  * Sends an unit into the wait status
155  */
156 actions.wait = {
157   key: "wait",
158 
159   condition: function(x,y,rStP,rStU,selected,property,unit){
160 
161     if( selected === null || !game.isUnit(selected) ) return false;
162     return ( rStU === game.RELATIONSHIP_NONE ||
163              rStU === game.RELATIONSHIP_SAME_OBJECT );
164   },
165 
166   action: function( x, y, suid ){
167     game.markAsUnusable( suid );
168   }
169 };
