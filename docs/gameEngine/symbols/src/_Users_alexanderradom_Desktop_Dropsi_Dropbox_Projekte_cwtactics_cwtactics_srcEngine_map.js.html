135 /**
136  * Extracts the identical number from an unit object.
137  *
138  * @param unit
139  */
140 game.extractUnitId = function( unit ){
141   if( unit === null ){
142     util.logError("unit argument cannot be null");
143   }
144 
145   var units = domain.units;
146   for( var i=0,e=units.length; i<e; i++ ){
147     if( units[i] === unit ) return i;
148   }
149 
150   // illegal unit object ?!
151   util.logError("cannot find unit",util.objectToJSON(unit));
152 };

205 /**
206  * Returns the relationship between two player identicals.
207  *
208  * @param pidA player id or ownable object
209  * @param pidB player id or ownable object
210  */
211 game.relationshipBetween = function( pidA, pidB ){
212   if( pidA === null || pidB === null ){
213     return game.RELATIONSHIP_NONE;
214   }
215 
216   if( typeof pidA !== 'number' &&  typeof pidB !== 'number'
217     && pidA === pidB ) return game.RELATIONSHIP_SAME_OBJECT;
218 
219   if( typeof pidA !== 'number' ) pidA = pidA.owner;
220   if( typeof pidB !== 'number' ) pidB = pidB.owner;
221 
222   if( pidA === pidB ) return game.RELATIONSHIP_SAME_OWNER;
223   else {
224     var tidA = game.player( pidA );
225     var tidB = game.player( pidB );
226     if( tidA === tidB ){
227       return game.RELATIONSHIP_ALLIED;
228     }
229     else return game.RELATIONSHIP_ENEMY;
230   }
231 };

242 /**
243  * Sets the position of an unit.
244  *
245  * @param uid
246  * @param tx
247  * @param ty
248  */
249 game.setUnitPosition = function( uid, tx, ty ){
250   var unit = game.unitById(uid);
251   var ox = unit.x;
252   var oy = unit.y;
253   var uPosMap = domain.unitPosMap;
254 
255   // clear old position
256   uPosMap[ox][oy] = null;
257 
258   // unit has a new position
259   if( arguments.length > 1 ){
260     unit.x = tx;
261     unit.y = ty;
262 
263     uPosMap[tx][ty] = unit;
264   }
265 };

285 /**
286  * One unit joins another unit and combines its health points with the other
287  * unit to heal damages. The joining unit will be removed.
288  */
289 actions.join = {
290 
291   condition: function( x,y,rStP,rStU,selected,property,unit, moved ){
292     if( selected === null ) return;
293     if( rStU !== game.RELATIONSHIP_SAME_OWNER ) return;
294 
295     return ( selected.type === unit.type && unit.hp < 99 );
296   },
297 
298   action: function( x, y, suid, tprid, tuid ){
299     var unit = game.unitById(suid);
300     var junit = game.unitById(tuid);
301     game.eraseUnitPosition( suid );
302     unit.owner = -1;
303 
304     junit.hp += unit.hp;
305     if( junit.hp > 99 ) junit.hp = 99;
306 
307     signal.emit( signal.EVENT_INTO_WAIT, tuid );
308   }
309 };

312 /**
313  * Loads a map and initializes the game context.
314  *
315  * @event map.load
316  */
317 signal.connect( game.EVENT_LOAD_MAP, function( channel, mapData ){
318   if( util.DEBUG ){
319     util.logInfo("start loading map");
320   }
321 
322   // TODO map is data from outside, check it via shema
323 
324   // meta data
325   domain.mapWidth = mapData.width;
326   domain.mapHeight = mapData.height;
327 
328   // TODO this should be in map data possible
329   // TODO too ( e.g. if map is the result of a save game )
330   // this._day = 0;
331   // this._activePlayer = 0;
332   domain.turnOwner = 0;
333 
334   // filler
335   for( var x=0, e1= domain.mapWidth; x<e1; x++ ){
336     for( var y=0, e2= domain.mapHeight; y<e2; y++ ){
337       domain.map[x][y] = mapData.filler;
338     }
339   }
340 
341   for( var x=0, e1= mapData.mapWidth; x<e1; x++ ){
342     for( var y=0, e2= mapData.maxHeight; y<e2; y++ ){
343       domain.unitPosMap[x][y] = null;
344     }
345   }
346 
347   // special tiles
348   var cols = Object.keys( mapData.data );
349   for( var i=0,e=cols.length; i<e; i++ ){
350 
351     var x = parseInt(cols[i],10);
352     var rows = Object.keys( mapData.data[ cols[i] ] );
353     for( var j=0,f=rows.length; j<f; j++ ){
354 
355       var y = parseInt(rows[j],10);
356       domain.map[x][y] = mapData.data[ cols[i] ][ rows[j] ];
357     }
358   }
359 
360   for( var i=0, e= domain.units.length; i<e; i++){
361     domain.units[i].owner = CWT_INACTIVE_ID;
362   }
363 
364   // players
365   for( var i = 0, e = mapData.players.length; i<e; i++){
366     var s_player = mapData.players[i];
367     var t_player = domain.players[i];
368 
369     t_player.gold = s_player.gold;
370     t_player.team = s_player.team;
371     t_player.name = s_player.name;
372 
373     // units
374     var startIndex = i*CWT_MAX_UNITS_PER_PLAYER;
375     for( var j = 0, ej = s_player.units.length; j<ej; j++){
376       var s_unit = s_player.units[j];
377       var t_unit = domain.units[ startIndex+j ];
378 
379       t_unit.fuel  = s_unit.fuel;
380       t_unit.x     = s_unit.x;
381       t_unit.y     = s_unit.y;
382       t_unit.hp    = s_unit.hp;
383       t_unit.ammo  = s_unit.ammo;
384       t_unit.type  = s_unit.type;
385       t_unit.owner = s_unit.owner;
386       t_unit.loadedIn = -1;
387 
388       // TODO use identical
389       domain.unitPosMap[s_unit.x][s_unit.y] = t_unit;
390     }
391   }
392   for( var i=mapData.players.length,e= domain.players.length; i<e; i++){
393     if( i !== game.neutralPlayerId() ){
394       domain.players[i].team = CWT_INACTIVE_ID;
395     }
396   }
397 
398   // properties
399   for( var i = 0, e = mapData.properties.length; i<e; i++){
400     var s_property = mapData.properties[i];
401     var t_property = domain.properties[i];
402 
403     t_property.owner          = s_property.owner;
404     if( t_property.owner === undefined ) t_property.owner = game.neutralPlayerId();
405 
406     t_property.capturePoints  = s_property.capturePoints;
407     t_property.type           = s_property.type;
408     t_property.x              = s_property.x;
409     t_property.y              = s_property.y;
410 
411     // TODO use identical
412     domain.propertyPosMap[t_property.x][t_property.y] = t_property;
413   }
414   for( var i=mapData.properties.length,e=domain.properties.length; i<e; i++){
415     domain.properties[i].owner = CWT_INACTIVE_ID;
416   }
417 
418   // add actors
419   var startIndex= domain.turnOwner*CWT_MAX_UNITS_PER_PLAYER;
420   for( var i= startIndex,
421            e= i+CWT_MAX_UNITS_PER_PLAYER; i<e; i++ ){
422 
423     domain.leftActors[i-startIndex] = (game.unitById(i) !== null)? true : false;
424   }
425 
426   if( util.DEBUG ){
427     util.logInfo("map successfully loaded");
428   }
429 });
