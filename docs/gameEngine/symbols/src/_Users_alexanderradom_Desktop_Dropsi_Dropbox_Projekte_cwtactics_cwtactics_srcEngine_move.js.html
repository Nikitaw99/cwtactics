 12 game.moveCheck = function( tx, ty, mvp, mType, checkMap, player,
 13                           getTileVal, setTileVal ){
 14 
 15   var cost = game.moveCosts( mType, domain.map[ tx ][ ty ] );
 16   if( cost != 0 ){
 17 
 18     // TODO SUPPORT MOVE THROUGH ALLIED AND OWN UNITS
 19     var unit = game.unitByPos( tx, ty );
 20     if( unit !== null && game.player(unit.owner).team !== player.team ){
 21       return; }
 22 
 23     var rest = mvp-cost;
 24     if( rest >= 0 ){
 25 
 26       if( getTileVal( tx,ty ) < rest ){
 27 
 28         // ADD TO TO CHECK MAP
 29         checkMap.push( [ tx, ty, rest ] );
 30 
 31         // ADD TO MOVE MAP
 32         setTileVal( tx,ty, rest );
 33 
 34         // NEGATIVE COSTS MEANS SAME COSTS BUT NOT USABLE AS TARGET
 35         //if( unit !== null && cwt.data.player(unit.owner).team === player.team ){
 36         // movemap[ tx ][ ty ] = -rest; }
 37       }
 38     }
 39   }
 40 };

 66 /**
 67  * Returns the costs for a movetype to move onto a tile type.
 68  *
 69  * @param movetype
 70  * @param tiletype
 71  */
 72 game.moveCosts = function( movetype, tiletype ){
 73   var c;
 74 
 75   // search id
 76   c = movetype.costs[ tiletype ];
 77 
 78   if( c === undefined ) c = movetype.costs["*"];
 79 
 80   return c;
 81 };

131 /**
132  * Sets the move map with the movecosts in a table ( array of arrays ).
133  *
134  */
135 game.prepareMovepathSelection = function( uid, getTileVal, setTileVal ){
136 
137   var unit   = game.unitById( uid );
138   var type   = game.unitSheet( unit.type );
139   var mType  = game.movetypeSheet( type.moveType );
140   var player = game.player( unit.owner );
141   var range  = type.moveRange;
142   var x = unit.x;
143   var y = unit.y;
144 
145   // DECREASE RANGE IF NOT ENOUGH FUEL IS AVAILABLE
146   if( unit.fuel < range ) range = unit.fuel;
147 
148   // ADD START TILE TO MAP
149   setTileVal( x, y, range );
150 
151   // BUILD MOVE MAP
152   var tile, cost, rest;
153   var needsCheck = [ [ x, y, range ] ]; //TODO REFA -> FLAT ARRAY
154   while( needsCheck.length > 0 ){
155 
156     // GET NEXT TILE
157     tile = needsCheck[ 0 ];
158     needsCheck.splice(0,1);
159 
160     // UP
161     if( tile[1] > 0 )
162       game.moveCheck( tile[0], tile[1]-1, tile[2], mType, needsCheck, player,
163         getTileVal, setTileVal );
164 
165     // RIGHT
166     if( tile[0] < domain.mapWidth-1 )
167       game.moveCheck( tile[0]+1, tile[1], tile[2], mType, needsCheck, player,
168         getTileVal, setTileVal  );
169 
170     // DOWN
171     if( tile[1] < domain.mapHeight-1 )
172       game.moveCheck( tile[0], tile[1]+1, tile[2], mType, needsCheck, player,
173         getTileVal, setTileVal  );
174 
175     // LEFT
176     if( tile[0] > 0 )
177       game.moveCheck( tile[0]-1, tile[1], tile[2], mType, needsCheck, player,
178         getTileVal, setTileVal  );
179   }
180 
181   // CONVERT LEFT MOVE POINTS TO COST POINTS
182   var moveMapDim = CWT_MAX_MOVE_RANGE*2+1;
183   for( var x=0,xe=moveMapDim; x<xe; x++ ){
184     for( var y=0,ye=moveMapDim; y<ye; y++ ){
185 
186       if( getTileVal(x,y) !== -1 ){
187 
188         var cost = game.moveCosts( mType, domain.map[x][y] );
189         setTileVal(x,y,cost);
190       }
191     }
192   }
193 };

195 /**
196  * Moves an unit from A to B with a given path.
197  *
198  * @param uid unit identical number
199  * @param sx x coordinate of the start point
200  * @param sy y coordinate of the start point
201  * @param way array of move codes
202  */
203 game.moveUnitFromAtoB = function( uid, sx, sy, way ){
204 
205   var cX = sx;
206   var cY = sy;
207   var unit = game.unitById( uid );
208   var uType = game.unitSheet( unit.type );
209   var mType = game.movetypeSheet( uType.moveType );
210 
211   // CHECK MOVE WAY END
212   var lastIndex = way.length-1;
213   var fuelUsed = 0;
214   for( var i=0,e=way.length; i<e; i++ ){
215 
216     // GET NEW CURRENT POSITION
217     switch( way[i] ){
218 
219       case game.MOVE_CODE_UP:
220         if( cY === 0 ) util.logError(
221           "cannot do move command UP because",
222           "current position is at the border"
223         );
224         cY--;
225         break;
226 
227       case game.MOVE_CODE_RIGHT:
228         if( cX === domain.mapWidth-1 ) util.logError(
229           "cannot do move command RIGHT because",
230           "current position is at the border"
231         );
232         cX++;
233         break;
234 
235       case game.MOVE_CODE_DOWN:
236         if( cY === domain.mapHeight-1 )util.logError(
237           "cannot do move command DOWN because",
238           "current position is at the border"
239         );
240         cY++;
241         break;
242 
243       case game.MOVE_CODE_LEFT:
244         if( cX === 0 ) util.logError(
245           "cannot do move command LEFT because",
246           "current position is at the border"
247         );
248         cX--;
249         break;
250 
251       default: util.logError("unknown command ",way[i]);
252     }
253 
254     // IS WAY BLOCKED ?
255     if( game.isWayBlocked( cX, cY, unit.owner, i == e-1 ) ){
256 
257       lastIndex = i-1;
258 
259       // GP BACK
260       switch( way[i] ){
261 
262         case game.MOVE_CODE_UP:
263           cY++;
264           break;
265 
266         case game.MOVE_CODE_RIGHT:
267           cX--;
268           break;
269 
270         case game.MOVE_CODE_DOWN:
271           cY--;
272           break;
273 
274         case game.MOVE_CODE_LEFT:
275           cX++;
276           break;
277       }
278 
279 
280       if( lastIndex == -1 ){
281 
282         // THAT IS A FAULT
283         cwt.error(
284           "unit is blocked by an enemy, but the enemy",
285           "stands beside the start tile, that is a logic fault!"
286         );
287       }
288 
289       break;
290     }
291 
292     // INCREASE FUEL USAGE
293     fuelUsed += game.moveCosts( mType, domain.map[cX][cY] );
294   }
295 
296   unit.fuel -= fuelUsed;
297 
298   game.eraseUnitPosition( uid );
299   // DO NOT SET NEW POSITION IF THE POSITION IS OCCUPIED
300   // THE SET POSITION LOGIC MUST BE DONE BY THE ACTION
301   if( game.unitByPos(cX,cY) === null ){
302       game.setUnitPosition( uid, cX, cY );
303   }
304 
305   if( util.DEBUG ){
306     util.logInfo(
307       "moved unit",uid,
308       "from (",sx,",",sy,")",
309       "to (",cX,",",cY,")"
310     );
311   }
312 };
