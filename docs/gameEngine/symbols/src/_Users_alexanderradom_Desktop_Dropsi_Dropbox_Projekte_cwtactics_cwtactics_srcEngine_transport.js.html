1 /**
  2  * Retusn a list of loaded unit ids by a given transporter id.
  3  *
  4  * @param tid
  5  */
  6 game.getLoadedIds = function( tid ){
  7   var loaded = [];
  8   var pid = game.unitById( tid ).owner;
  9   for( var i=CWT_MAX_UNITS_PER_PLAYER*pid,
 10          e=i+CWT_MAX_UNITS_PER_PLAYER; i<e; i++ ){
 11 
 12     if( i !== tid ){
 13       var unit = game.unitById( i );
 14       if( unit !== null && unit.loadedIn === tid ){
 15         loaded.push( i );
 16       }
 17     }
 18   }
 19 
 20   return loaded;
 21 };

 23 /**
 24  * Has a transporter unit with id tid loaded units? Returns true if yes, else
 25  * false.
 26  *
 27  * @param tid
 28  */
 29 game.hasLoadedIds = function( tid ){
 30   var pid = game.unitById( tid ).owner;
 31   for( var i=CWT_MAX_UNITS_PER_PLAYER*pid,
 32          e=i+CWT_MAX_UNITS_PER_PLAYER; i<e; i++ ){
 33 
 34     if( i !== tid ){
 35       var unit = game.unitById( i );
 36       if( unit !== null && unit.loadedIn === tid ){
 37         return true;
 38       }
 39     }
 40   }
 41 
 42   return false;
 43 };

 45 /**

 56 /**
 57  * Loads the unit with id lid into a tranporter with the id tid.
 58  *
 59  * @param lid
 60  * @param tid
 61  */
 62 game.loadUnitInto = function( lid, tid ){
 63   if( !game.canLoad( lid,tid ) ){
 64     util.logError("transporter unit",tid,"cannot load unit",lid);
 65   }
 66 
 67   game.unitById( lid ).loadedIn = tid;
 68 };

 70 /**
 71  * Unloads the unit with id lid from a tranporter with the id tid.
 72  *
 73  * @param lid
 74  * @param tid
 75  */
 76 game.unloadUnitFrom = function( lid, tid ){
 77   game.unitById( lid ).loadedIn = -1;
 78 };

 80 /**
 81  * Returns true if a tranporter with id tid can load the unit with the id
 82  * lid. This function also calculates the resulting weight if the transporter
 83  * would load the unit. If the calculated weight is greater than the maxiumum
 84  * loadable weight false will be returned.
 85  *
 86  * @param lid
 87  * @param tid
 88  */
 89 game.canLoad = function( lid, tid ){
 90   var tp = game.unitById(tid);
 91   var lu = game.unitById(lid);
 92 
 93   // CALCULATE CURRENT LOADED WEIGHT
 94   var cW = 0;
 95   for( var i=CWT_MAX_UNITS_PER_PLAYER*tp.owner,
 96          e=i+CWT_MAX_UNITS_PER_PLAYER; i<e; i++ ){
 97 
 98     if( i !== tid ){
 99       var unit = game.unitById( i );
100       if( unit !== null && unit.loadedIn === tid ){
101         cW += game.unitSheet( game.unitById(i).type ).weight;
102       }
103     }
104   }
105 
106   // CALCULATE FUTURE WEIGHT
107   cW += game.unitSheet( lu.type ).weight;
108 
109   var tps = game.unitSheet( tp.type ).transport;
110   if( cW > tps.maxWeight ) return false;
111 
112   // IS UNIT TYPE LOADABLE
113   var unitSh = game.unitSheet( game.unitById(lid).type );
114   var tpsL = tps.canLoad;
115 
116   // ID
117   if( tpsL.indexOf( game.unitById(lid).type ) !== -1 ) return true;
118 
119   // TAGS
120   var tags = unitSh.tags;
121   for( var i=0,e=tags.length; i<e; i++ ){
122     if( tpsL.indexOf( tags[i] ) !== -1 ) return true;
123   }
124 
125   // ALL TYPE
126   if( tpsL.indexOf("*") !== -1 ) return true;
127 
128   return false;
129 };

131 /**
132  * Returns true if the unit with id tid is a traensporter, else false.
133  *
134  * @param tid

136 game.isTransport = function( tid ){

140 /**
141  * Loads an unit into a transporter unit.
142  */
143 actions.loadUnit = {
144   condition: function( x,y,rStP,rStU,selected,property,unit, moved ){
145     if( selected === null ) return;
146     if( rStU !== game.RELATIONSHIP_SAME_OWNER ) return;
147 
148     return (
149       game.isTransport( game.extractUnitId(unit) ) &&
150         game.canLoad( game.extractUnitId(selected), game.extractUnitId(unit) )
151       );
152   },
153 
154   action: function( x, y, suid, tprid, tuid ){
155     game.eraseUnitPosition( suid );
156     game.loadUnitInto( suid, tuid );
157     game.setUnitPosition( tuid, x,y );
158     signal.emit( signal.EVENT_INTO_WAIT, suid );
159   }
160 };

162 /**
163  * Unloads an unit from a tranporter unit.
164  */
165 actions.unloadUnit = {
166 
167   targetSelection: true,
168   hasSubMenu: true,
169 
170   prepareSubMenu: function( menu, selected, tx, ty ){
171     var loads = game.getLoadedIds( game.extractUnitId( selected ) );
172     for( var i=0,e=loads.length; i<e; i++ ){
173       menu.push( loads[i] );
174     }
175   },
176 
177   prepareTargets: function( selected, tx, ty, subEntry ){
178     var load = game.unitById( subEntry );
179     var loadS = game.unitSheet( load.type );
180     var loadMvS = game.movetypeSheet( loadS.moveType );
181 
182     if( tx > 0 ){
183       if( game.unitByPos( tx-1, ty ) === null &&
184         game.moveCosts( loadMvS, game.tileByPos(tx-1,ty) ) !== -1  ){
185         controller.setValueOfSelectionPos( tx-1,ty,1 );
186       }
187     }
188 
189     if( ty > 0 ){
190       if( game.unitByPos( tx, ty-1 ) === null &&
191         game.moveCosts( loadMvS, game.tileByPos(tx,ty-1) ) !== -1  ){
192         controller.setValueOfSelectionPos( tx,ty-1,1 );
193       }
194     }
195 
196     if( ty < game.mapHeight()-1 ){
197       if( game.unitByPos( tx, ty+1 ) === null &&
198         game.moveCosts( loadMvS, game.tileByPos(tx,ty+1) ) !== -1  ){
199         controller.setValueOfSelectionPos( tx,ty+1,1 );
200       }
201     }
202 
203     if( tx < game.mapWidth()-1 ){
204       if( game.unitByPos( tx+1, ty ) === null &&
205         game.moveCosts( loadMvS, game.tileByPos(tx+1,ty) ) !== -1  ){
206         controller.setValueOfSelectionPos( tx+1,ty,1 );
207       }
208     }
209   },
210 
211   condition: function( x,y,rStP,rStU,selected,property,unit, moved ){
212     if( selected === null || !game.isUnit(selected) ) return;
213     if( rStU !== game.RELATIONSHIP_NONE ) return;
214 
215     return (
216       game.isTransport( game.extractUnitId(selected) ) &&
217         game.hasLoadedIds( game.extractUnitId(selected) )
218       );
219   },
220 
221   action: function( x, y, sid, prid, tid, ax, ay, subEntry ){
222     game.unloadUnitFrom( sid, tid );
223     game.setUnitPosition( subEntry, ax, ay );
224     signal.emit( signal.EVENT_INTO_WAIT, sid );
225     signal.emit( signal.EVENT_INTO_WAIT, subEntry );
226   }
227 };
