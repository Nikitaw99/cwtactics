 42 /**
 43  * Loads a modification into the game engine.
 44  */
 45 persistence.loadModification = function( modification ){
 46   var list;
 47 
 48   list = modification.movetypes;
 49   for( var i=0,e=list.length; i<e; i++ ){
 50     sheets.parseSheet( list[i], sheets.MOVE_TYPE_SHEET );
 51   }
 52 
 53   list = modification.weapons;
 54   for( var i=0,e=list.length; i<e; i++ ){
 55     sheets.parseSheet( list[i], sheets.WEAPON_TYPE_SHEET );
 56   }
 57 
 58   list = modification.tiles;
 59   for( var i=0,e=list.length; i<e; i++ ){
 60     sheets.parseSheet( list[i], sheets.TILE_TYPE_SHEET );
 61   }
 62 
 63   list = modification.units;
 64   for( var i=0,e=list.length; i<e; i++ ){
 65     sheets.parseSheet( list[i], sheets.UNIT_TYPE_SHEET );
 66   }
 67 
 68   var langs = modification.locale;
 69   if( langs !== undefined ){
 70 
 71     var langIds = Object.keys( langs );
 72     for( var i=0,e=langIds.length; i<e; i++ ){
 73       locale.appendToLanguage( langIds[i], langs[ langIds[i] ] );
 74     }
 75   }
 76 };
