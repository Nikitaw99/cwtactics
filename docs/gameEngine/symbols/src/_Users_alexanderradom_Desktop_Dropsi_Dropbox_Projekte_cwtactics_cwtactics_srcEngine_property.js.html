 18 /**
 19  * Returns a property by its id.
 20  *
 21  * @param id
 22  */
 23 game.propertyById = function(id){
 24   if( id < 0 || domain.properties.length <= id ){
 25     util.logError("invalid property id",id);
 26   }
 27 
 28   var o = domain.properties[id];
 29   return ( o.owner === CWT_INACTIVE_ID )? null : o;
 30 };

 32 /**
 33  * Returns a property by its position.
 34  *
 35  * @param id
 36  */
 37 game.propertyByPos = function( x, y ){
 38   return domain.propertyPosMap[x][y];
 39 };

 41 /**
 42  *
 43  * @param x
 44  * @param y
 45  */
 46 game.tileIsProperty = function( x,y ){
 47   var prop = game.propertyByPos(x,y);
 48   if( prop === null ) return false;
 49   else{
 50     return game.extractPropertyId( prop );
 51   }
 52 };

 54 /**
 55  * Extracts the identical number from a property object.
 56  *
 57  * @param unit
 58  */
 59 game.extractPropertyId = function( property ){
 60   if( property === null ){
 61     util.logError("property argument cannot be null");
 62   }
 63 
 64   var props = domain.properties;
 65   for( var i=0,e=props.length; i<e; i++ ){
 66     if( props[i] === property ) return i;
 67   }
 68 
 69   // illegal unit object ?!
 70   util.logError("cannot find property",property );
 71 };

 73 /**
 74  * Captures a property by a given unit.
 75  *
 76  * @param x x position
 77  * @param y y position
 78  * @param id identical number for the capturing unit object
 79  */
 80 game.captureProperty = function( x, y, id ){
 81   var unit   = game.unitById(id);
 82   var unitSh = game.unitSheet( unit.type );
 83   var prop   = game.propertyByPos( x, y );
 84   prop.capturePoints -= unitSh.captures;
 85   if( prop.capturePoints <= 0 ){
 86     if( util.DEBUG ){
 87       util.logInfo( "property at (",x,",",y,") captured");
 88     }
 89 
 90     if( prop.type === 'HQ' ){
 91       signal.emit( signal.EVENT_HQ_LOST,
 92         prop.owner, game.extractPropertyId(prop) );
 93     }
 94 
 95     // set new meta data for property
 96     prop.capturePoints = 20;
 97     prop.owner = unit.owner;
 98   }
 99 };

101 /** @constant */
102 signal.EVENT_HQ_LOST = "capture:hqLost";
103 
104 signal.connect( signal.EVENT_HQ_LOST, function( ch, pid, hqid ){
105   var oldPlayer = game.player( pid );
106 
107   for( var i = pid * CWT_MAX_UNITS_PER_PLAYER,
108            e = i+CWT_MAX_UNITS_PER_PLAYER; i<e; i++ ){
109 
110     domain.units[i].team = -1;
111     domain.unitPosMap[ domain.units[i].x ][ domain.units[i].y ] = null;
112   }
113 
114   for( var i = 0, e = domain.properties.length; i<e; i++ ){
115     if( domain.properties[i].owner === pid ){
116         domain.properties[i].owner = -1;
117     }
118   }
119 
120   oldPlayer.team = -1;
121 
122   // check win/loose
123   if( game.annithilationWinner() ){
124     if( util.DEBUG ){
125       util.logInfo("the game ends because no opposite players exists");
126     }
127   }
128 });

152 actions.silofire = {
153 
154   targetSelection: true,
155   freeTarget: true,
156 
157   _doDamage: function( x,y, dmg ){
158     var unit = game.unitByPos(x,y);
159     if( unit !== null ){
160       unit.hp -= dmg;
161       if( unit.hp < 9 ) unit.hp = 9;
162     }
163   },
164 
165   condition: function( x,y,rStP,rStU,selected,property,unit, moved ){
166     // if( property.owner !== game.getTurnOwner() ) return;
167     if( selected === null ) return false;
168     if( selected.type !== "INFT_OS" && selected.type !== "MECH_OS" ){
169       return false;
170     }
171 
172     return ( property.type === "SILO" );
173   },
174 
175   action: function( x, y, suid, tprid, tuid, ax, ay ){
176     var mapW = game.mapWidth();
177     var mapH = game.mapHeight();
178     var dmgF = actions.silofire._doDamage;
179     var dmg  = 20;
180 
181     // LEFT
182     if( ax-1 >= 0 ){
183 
184       // OUTER LEFT COLUMN
185       if( ax-2 >= 0 )    dmgF( ax-2, ay, dmg );
186 
187       // LEFT COLUMN
188       if( ay-1 >= 0 )    dmgF( ax-1, ay-1, dmg );
189       dmgF( ax-1, ay,   dmg );
190       if( ay+1 < mapH )  dmgF( ax-1, ay+1, dmg );
191     }
192 
193     // X COLUMN
194     if( ay-1 >= 0 ){     dmgF( ax, ay-1, dmg );
195       if( ay-2 >= 0 ){   dmgF( ax, ay-2, dmg ); }}
196     dmgF( ax, ay,   dmg );
197     if( ay+1 < mapH ){   dmgF( ax, ay+1, dmg );
198       if( ay+2 < mapH ){ dmgF( ax, ay+2, dmg ); }}
199 
200     // RIGHT
201     if( ax+1 < mapW ){
202 
203       // RIGHT COLUMN
204       if( ay-1 >= 0 )    dmgF( ax+1, ay-1, dmg );
205       dmgF( ax+1, ay,   dmg );
206       if( ay+1 < mapH )  dmgF( ax+1, ay+1, dmg );
207 
208       // OUTER RIGHT COLUMN
209       if( ax+2 < mapW )  dmgF( ax+2, ay, dmg );
210     }
211 
212     // SET EMPTY TYPE
213     game.propertyById( tprid ).type = "SILO_EMPTY";
214     signal.emit( signal.EVENT_INTO_WAIT, suid );
215   }
216 };
