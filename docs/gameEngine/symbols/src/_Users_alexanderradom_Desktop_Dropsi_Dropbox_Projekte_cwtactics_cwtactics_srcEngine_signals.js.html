1 /**
  2  * @private
  3  */
  4 signal._data = {};
  5 
  6 /**
  7  * Connects a listener to a channel. This function will be invoked if data will
  8  * be pushed through the channel by @link{signal.emit}.
  9  *
 10  * @param channel
 11  * @param cb
 12  */
 13 signal.connect = function( channel, cb ){
 14   var data = signal._data;
 15 
 16   // CREATE CHANNEL IF IT DOES NOT EXISTS
 17   if( !data.hasOwnProperty(channel) ) data[channel] = [];
 18 
 19   // PUSH CALLBACK
 20   data[channel].push( cb );
 21 };
 22 
 23 /**
 24  * Removes a listener from a channel. Returns true if the callback was removed
 25  * from a channel, else false.
 26  *
 27  * @param channel
 28  * @param cb
 29  */
 30 signal.disconnect = function( channel, cb ){
 31   var data = signal._data;
 32 
 33   if( !data.hasOwnProperty(channel) ) return false;
 34 
 35   var cbList = data[channel];
 36   for( var i=0,e=cbList.length; i<e; i++ ){
 37     if( cbList[i] === cb ){
 38 
 39       // CALLBACK FOUND, REMOVE IT
 40       cbList.splice(i,1);
 41       return true;
 42     }
 43   }
 44 
 45   // CALLBACK NOT FOUND
 46   return false;
 47 };
 48 
 49 /**
 50  * Pushes data into a channel. All connected listeners will be invoked with
 51  * the given arguments.
 52  *
 53  * @param channel
 54  */
 55 signal.emit = function( channel ){
 56   var data = signal._data;
 57 
 58   // CREATE CHANNEL IF IT DOES NOT EXISTS
 59   if( !data.hasOwnProperty(channel) ) return;
 60 
 61   var cbList = data[channel];
 62   for( var i=0,e=cbList.length; i<e; i++ ){
 63 
 64     // INVOKE CALLBACK
 65     cbList[i].apply( null, arguments );
 66   }
 67 };
