    Creates a CircularBuffer with a given size. All slots will be filled with objects of the given clazz. The constructor of the given clazz will be called with zero arguments.

    exports.createBufferByClass = function (clazz, size) {
  var list = new exports.CircularBuffer(size);

  while (size > 0) {
    list.push(new clazz());
    size--;
  }

  return list;
};
