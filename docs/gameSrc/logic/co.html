    exports.activatePower = function (player, level) {
  if (constants.DEBUG) {
    assert(player instanceof model.Player && (level === exports.POWER_LEVEL_COP || level === exports.POWER_LEVEL_SCOP));
  }

  player.power = 0;
  player.activePower = level;
  player.powerUsed++;
};
